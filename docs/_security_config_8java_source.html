9/**
10 * @brief Configures Spring Security for role-based URL access control.
11 *
12 * Authentication is handled entirely by Firebase (see {@link FirebaseConfig}).
13 * Spring Security's job here is authorisation only — it checks the role stored in the
14 * HTTP session and either lets the request through or redirects to the appropriate
15 * login page.
16 *
17 * CSRF, form-login, and HTTP-Basic are all disabled because the app uses Firebase ID
18 * tokens and session attributes for auth state rather than Spring's built-in mechanisms.

31 /**
32 * Declares the security filter chain that governs every HTTP request.
33 *
34 * Access rules summary:
35 * - Public pages (site, assets, login endpoints) — {@code permitAll()}
36 * - {@code /patient/**} — {@code ROLE_PATIENT} only
37 * - {@code /medical-specialist/**} — {@code ROLE_MEDICALSPECIALIST} only
38 * - {@code /secretary/**} — {@code ROLE_SECRETARY} only
39 * - {@code /buyer/**} — {@code ROLE_BUYER} only
40 * - {@code /technician/**} — {@code ROLE_TECHNICIAN} only
41 * - {@code /staff-manager/**} — {@code ROLE_STAFFMANAGER} only
42 * - {@code /payment/**} — patient or secretary
43 * - {@code /employee/resources/**} — any staff role
44 *
45 * On auth failure: patient paths redirect to {@code /patient/login}, all others to {@code /login}.
46 *
47 * @param http Spring Security's builder for the filter chain.
48 * @return The configured {@link SecurityFilterChain}.
49 * @throws Exception When any security builder call fails.
50 */
51 @Bean
52 public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
53 http
54 .authorizeHttpRequests(auth -> auth
55 // ── Public pages ─────────────────────────────────────────
56 .requestMatchers(
57 "/", "/login", "/logout",
58 "/patient/login", "/patient/authenticate", "/patient/logout",
59 "/patient/spid-callback", "/patient/cie-callback",
60 "/mock-spid/login", "/mock-spid/authenticate",
61 "/mock-cie/login", "/mock-cie/authenticate",
62 "/authenticate", "/select-role",
63 "/departments/**", "/questionnaire/**",
64 "/specialists/**", "/doctors",
65 "/about", "/contact", "/map/**",
66 "/privacy", "/gdpr", "/terms", "/accessibility",
67 "/images/**", "/css/**", "/js/**",
68 "/webjars/**", "/error"
69 ).permitAll()
70 // ── Patient ──────────────────────────────────────────────
71 .requestMatchers("/patient/**").hasRole("PATIENT")
72 // ── Staff ────────────────────────────────────────────────
73 .requestMatchers("/medical-specialist/**").hasRole("MEDICALSPECIALIST")
74 .requestMatchers("/secretary/**").hasRole("SECRETARY")
75 .requestMatchers("/buyer/**").hasRole("BUYER")
76 .requestMatchers("/technician/**").hasRole("TECHNICIAN")
77 .requestMatchers("/staff-manager/**").hasRole("STAFFMANAGER")
78 .requestMatchers("/payment/**").hasAnyRole("PATIENT", "SECRETARY")
79 .requestMatchers("/employee/resources/**").hasAnyRole(
80 "MEDICALSPECIALIST", "SECRETARY", "BUYER", "TECHNICIAN", "STAFFMANAGER")
81 // ── Anything else requires authentication ─────────────────
82 .anyRequest().authenticated()
83 )
84 .csrf(csrf -> csrf.disable())
85 .formLogin(form -> form.disable())
86 .httpBasic(basic -> basic.disable())
87 .exceptionHandling(ex -> ex
88 // Redirect unauthenticated requests to the correct login page
89 .authenticationEntryPoint((request, response, authException) -> {
90 String path = request.getRequestURI();
91 String base = request.getContextPath();
92 if (path.startsWith(base + "/patient/")) {
93 response.sendRedirect(base + "/patient/login");
94 } else {
95 response.sendRedirect(base + "/login");
96 }
97 })
98 // Wrong role → back to appropriate login
99 .accessDeniedHandler((request, response, accessDeniedException) -> {
100 String path = request.getRequestURI();
101 String base = request.getContextPath();
102 if (path.startsWith(base + "/patient/")) {
103 response.sendRedirect(base + "/patient/login");
104 } else {
105 response.sendRedirect(base + "/login");
106 }
107 })
108 );
109
110 return http.build();
111 }
