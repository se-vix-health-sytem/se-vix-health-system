126 /**
127 * Admit a patient to a room via a secretary (UC22 - Triage).
128 * Uses the Secretary domain method, which delegates to InternationRoom domain logic.
129 */
130 @Transactional
131 public void admitPatient(Secretary secretary, Long patientId, Long roomId) {
132 Patient patient = patientRepository.findById(patientId)
133 .orElseThrow(() -> new RuntimeException("Patient not found with id: " + patientId));
134
135 Room room = findById(roomId);
136
137 if (!(room instanceof InternationRoom)) {
138 throw new RuntimeException("Room " + roomId + " is not an inpatient room");
139 }
140
141 InternationRoom inpatientRoom = (InternationRoom) room;
142
143 if (inpatientRoom.getNFreeBeds() <= 0) {
144 throw new RuntimeException("No available beds in room " + inpatientRoom.getNumber());
145 }
146
147 try {
148 // Domain: secretary assigns patient to room via model methods
149 secretary.setPatientInRoom(inpatientRoom, patient);
150 roomRepository.save(inpatientRoom);
151
152 auditService.log("ADMIT_PATIENT", "Patient", String.valueOf(patientId),
153 "Patient admitted to room: " + inpatientRoom.getNumber());
154 } catch (Exception e) {
155 throw new RuntimeException("Failed to admit patient: " + e.getMessage(), e);
156 }
157 }
158
159 /**
160 * Dismiss a patient from a room via a secretary (UC23 - Patient Dismissal).
161 * Uses the Secretary domain method, which delegates to InternationRoom domain logic.
162 */
163 @Transactional
164 public void dismissPatient(Secretary secretary, Long patientId, Long roomId) {
165 Patient patient = patientRepository.findById(patientId)
166 .orElseThrow(() -> new RuntimeException("Patient not found with id: " + patientId));
167
168 Room room = findById(roomId);
169
170 if (!(room instanceof InternationRoom)) {
171 throw new RuntimeException("Room " + roomId + " is not an inpatient room");
172 }
173
174 InternationRoom inpatientRoom = (InternationRoom) room;
175
176 if (!inpatientRoom.hasPatient(patient)) {
177 throw new RuntimeException("Patient " + patientId + " is not in room " + inpatientRoom.getNumber());
178 }
179
180 try {
181 // Domain: secretary dismisses patient from room via model methods
182 secretary.dismissPatient(inpatientRoom, patient);
183 roomRepository.save(inpatientRoom);
184
185 auditService.log("DISMISS_PATIENT", "Patient", String.valueOf(patientId),
186 "Patient dismissed from room: " + inpatientRoom.getNumber());
187 } catch (Exception e) {
188 throw new RuntimeException("Failed to dismiss patient: " + e.getMessage(), e);
189 }
190 }

192 /**
193 * Find which room a patient is currently in
194 */
195 public InternationRoom findPatientRoom(Long patientId) {
196 Patient patient = patientRepository.findById(patientId)
197 .orElseThrow(() -> new RuntimeException("Patient not found: " + patientId));
198
199 return getAllInpatientRooms().stream()
200 .filter(room -> room.hasPatient(patient))
201 .findFirst()
202 .orElse(null);
203 }
