14/**
15 * Simulates payment processing for appointment fees (UC-Payment flow).
16 *
17 * In the current demo, every charge always succeeds and returns a randomly generated
18 * transaction ID. Completed transactions are stored in memory only — they do not
19 * survive a server restart. A real implementation would replace {@link #processPayment}
20 * with an actual payment-gateway call (Stripe, Braintree, etc.) and persist
21 * the result to the database.
22 *
23 * @see com.nvivx.vixhealthsystem.controllers.services.PaymentController
24 * @see com.nvivx.vixhealthsystem.dto.PaymentRequest
25 * @see com.nvivx.vixhealthsystem.dto.PaymentResponse
26 */
27@Service
28public class PaymentService {

55 /**

68 /**

70 *

72 * @return The stored {@link PaymentStatus}, or a NOT_FOUND status if the

83 /**
84 * Processes a simulated payment and returns a transaction result.
85 *
86 * Always succeeds in demo mode. Generates a {@code TXN-XXXXXXXX} transaction ID,
87 * stores the completed status in memory, and writes an audit entry.
88 * The 500 ms delay is intentional — it mimics a real gateway round-trip for the demo.
89 *
90 * @param request Payment details including appointment ID and amount.
91 * @return Response containing the transaction ID and SUCCESS status.
92 */
93 public PaymentResponse processPayment(PaymentRequest request) {
94 PaymentResponse response = new PaymentResponse();
95
96 String transactionId = "TXN-" + UUID.randomUUID().toString().substring(0, 8).toUpperCase();
97
98 response.setTransactionId(transactionId);
99 response.setStatus("SUCCESS");
100 response.setMessage("Payment processed successfully (Demo Mode)");
101
102 PaymentStatus status = new PaymentStatus();
103 status.setStatus("COMPLETED");
104 status.setTimestamp(LocalDateTime.now().toString());
105 paymentStatusMap.put(transactionId, status);
106
107 auditService.log("PROCESS_PAYMENT", "Appointment", String.valueOf(request.getAppointmentId()),
108 String.format("Payment of €%.2f processed. Transaction: %s", request.getAmount(), transactionId));
109
110 // Mimics real gateway latency so the demo UI doesn't feel instant
111 try { Thread.sleep(500); } catch (InterruptedException e) { Thread.currentThread().interrupt(); }
112
113 return response;
114 }
115
116 // =========================================================
117 // HELPERS

120 /** Returns a status object indicating the transaction ID was not found. */
121 private PaymentStatus createNotFoundStatus() {
122 PaymentStatus status = new PaymentStatus();
123 status.setStatus("NOT_FOUND");
124 status.setTimestamp(LocalDateTime.now().toString());
125 return status;
126 }
