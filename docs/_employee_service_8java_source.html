253 /**
254 * Creates a new employee record and a corresponding Firebase Authentication account.
255 *
256 * A temporary password ({@code ChangeMe123!}) is issued and stored in {@link DevCredentialStore}
257 * for demo visibility. The Firebase UID is written back to the entity before persisting.
258 *
259 * @param employee Transient employee entity with at least {@code email} set.
260 * @return The persisted {@link Employee} with a generated ID and Firebase UID.
261 * @throws RuntimeException When the employee has no email or the Firebase API call fails.
262 */
263 @Transactional
264 public Employee createEmployee(Employee employee) {
265 if (employee.getEmail() == null || employee.getEmail().isBlank()) {
266 throw new RuntimeException("Employee email is required to create Firebase account");
267 }
268
269 try {
270 String temporaryPassword = "ChangeMe123!";
271
272 String firebaseUid = firebaseAuthService.createUser(
273 employee.getEmail(),
274 temporaryPassword
275 );
276
277 employee.setFirebaseUid(firebaseUid);
278
279 log.info("[DEMO] Firebase account created — Employee: {} {} | Email: {} | UID: {}",
280 employee.getName(), employee.getSurname(), employee.getEmail(), firebaseUid);
281
282 devCredentialStore.store(
283 employee.getEmail(),
284 employee.getName() + " " + employee.getSurname(),
285 employee.getClass().getSimpleName(),
286 temporaryPassword
287 );
288
289 } catch (Exception e) {
290 String rootCause = (e.getCause() != null) ? e.getCause().getMessage() : e.getMessage();
291 throw new RuntimeException("Firebase error: " + rootCause, e);
292 }
293
294 Employee saved = employeeRepository.save(employee);
295
296 auditService.log(
297 "CREATE_EMPLOYEE",
298 "Employee",
299 String.valueOf(saved.getId()),
300 "Created employee: " + employee.getName() + " "
301 + employee.getSurname()
302 + " (Type: "
303 + employee.getClass().getSimpleName()
304 + ")"
305 );
306
307 return saved;
308 }

310 /**
311 * Applies a partial update to an existing employee, ignoring {@code null} fields.
312 *
313 * Only non-null fields in {@code updatedData} overwrite the persisted values, making
314 * it safe to call with sparse payloads (e.g., updating only a phone number).
315 *
316 * @param id ID of the employee to update.
317 * @param updatedData Carrier object; {@code null} fields are ignored.
318 * @return The updated and re-persisted {@link Employee}.
319 * @throws RuntimeException When no employee with {@code id} exists.
320 */
321 @Transactional
322 public Employee updateEmployee(Long id, Employee updatedData) {
323 Employee employee = findById(id);
324
325 if (updatedData.getName() != null) employee.setName(updatedData.getName());
326 if (updatedData.getSurname() != null) employee.setSurname(updatedData.getSurname());
327 if (updatedData.getEmail() != null) employee.setEmail(updatedData.getEmail());
328 if (updatedData.getPhoneNumber() != null) employee.setPhoneNumber(updatedData.getPhoneNumber());
329 if (updatedData.getBirthDate() != null) employee.setBirthDate(updatedData.getBirthDate());
330 if (updatedData.getBirthPlace() != null) employee.setBirthPlace(updatedData.getBirthPlace());
331 if (updatedData.getGender() != '\0') employee.setGender(updatedData.getGender());
332 if (updatedData.getHireDate() != null) employee.setHireDate(updatedData.getHireDate());
333 if (updatedData.getDepartment() != null) employee.setDepartment(updatedData.getDepartment());
334
335 Employee saved = employeeRepository.save(employee);
336
337 auditService.log(
338 "UPDATE_EMPLOYEE",
339 "Employee",
340 String.valueOf(id),
341 "Updated employee details"
342 );
343
344 return saved;
345 }

347 /**
348 * Generates a Firebase password-reset link for the employee and logs it to console.
349 *
350 * The link is also recorded in {@link DevCredentialStore} so it is not lost between

354 * @throws RuntimeException When the employee has no email or the Firebase API call fails.

356 public void requestEmployeePasswordReset(Long employeeId) {

358
359 if (employee.getEmail() == null || employee.getEmail().isBlank()) {

362

364 String resetLink =
365 firebaseAuthService.generatePasswordResetLink(employee.getEmail());

368 employee.getName(), employee.getSurname(), employee.getEmail(), resetLink);
369
370 // Mark on the dev board that this account's password is no longer the default
371 devCredentialStore.markResetTriggered(employee.getEmail());
372
373 auditService.log(
374 "REQUEST_PASSWORD_RESET",
375 "Employee",
376 String.valueOf(employee.getId()),
377 "Password reset requested for employee: "
378 + employee.getName() + " "
379 + employee.getSurname()
380 );
381
382 } catch (Exception e) {
383 throw new RuntimeException(
384 "Unable to generate password reset link",
385 e
386 );
387 }

412 /**
413 * Deletes an employee from both the database and Firebase Authentication.
414 *
415 * The Firebase account is removed first; if that fails the database record is not
416 * deleted to avoid orphaned auth entries. The dev credential entry is cleaned up
417 * regardless of Firebase success to keep the demo board tidy.
418 *
419 * @param id ID of the employee to delete.
420 * @throws RuntimeException When no employee exists with {@code id}, or when the
421 * Firebase deletion fails.
422 */
423 @Transactional
424 public void deleteEmployee(Long id) {
425 Employee employee = findById(id);
426 String employeeInfo = employee.getName() + " " + employee.getSurname();
427
428 if (employee.getFirebaseUid() != null) {
429 try {
430 firebaseAuthService.deleteUser(employee.getFirebaseUid());
431 } catch (Exception e) {
432 throw new RuntimeException(
433 "Unable to delete Firebase account",
434 e
435 );
436 }
437 }
438
439 if (employee.getEmail() != null) {
440 devCredentialStore.remove(employee.getEmail());
441 }
442
443 employeeRepository.delete(employee);
444
445 auditService.log(
446 "DELETE_EMPLOYEE",
447 "Employee",
448 String.valueOf(id),
449 "Deleted employee: " + employeeInfo
450 );
451 }
