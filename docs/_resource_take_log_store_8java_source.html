15/**
16 * @brief JSON-backed append log that records every resource takeout event performed by employees.
17 *
18 * Entries are stored in {@code src/main/resources/storage/resource-take-logs.json} and are
19 * sorted newest-first on retrieval. All public methods are {@code synchronized} to prevent
20 * concurrent write corruption from multi-threaded request handling.
21 *
22 * Used by {@link InventoryService#removeResourceFromStorage(com.nvivx.vixhealthsystem.model.person.employee.Employee, Long, int)}
23 * to maintain a separate, human-readable consumption audit trail alongside the main
24 * {@link com.nvivx.vixhealthsystem.service.AuditService} log.
25 *
26 * @see InventoryService
27 */
28@Service

35 /**

55 /**

58 * @param employeeId ID of the employee who took the resource.
59 * @param employeeName Display name of the employee.
60 * @param employeeRole Role class name (e.g., {@code "Technician"}).
61 * @param resourceId Catalogue ID of the taken resource.

66 public Entry(Long employeeId, String employeeName, String employeeRole,

92 /**

96 public ResourceTakeLogStore() {
97 this.mapper = new ObjectMapper();
98 this.mapper.registerModule(new JavaTimeModule());

100 init();

107 /**
108 * Creates the backing JSON file and its parent directories if they do not exist.
109 *
110 * @throws RuntimeException When the file cannot be created due to an I/O error.
111 */
112 private void init() {
113 File file = new File(path);
114 if (!file.exists()) {
115 try {
116 file.getParentFile().mkdirs();
117 mapper.writeValue(file, new ArrayList<Entry>());
118 } catch (IOException e) {
119 throw new RuntimeException("Could not create resource-take-logs.json", e);
120 }
121 }
122 }

125 // READ OPERATIONS
126 // =========================================================
127
128 /**
129 * Returns all log entries sorted newest-first.
130 *
131 * @return Mutable list of entries; empty when the log file is absent or empty.
132 */
133 public synchronized List<Entry> getAll() {
134 try {
135 File file = new File(path);
136 if (!file.exists() || file.length() == 0) return new ArrayList<>();
137 List<Entry> entries = mapper.readValue(file, new TypeReference<List<Entry>>() {});
138 entries.sort((a, b) -> b.takenAt != null && a.takenAt != null
139 ? b.takenAt.compareTo(a.takenAt) : 0);
140 return entries;
141 } catch (IOException e) {
142 return new ArrayList<>();
143 }
144 }
145
146 /**
147 * Returns all log entries for a specific employee, newest-first.
148 *
149 * @param employeeId ID of the employee whose takeout history is requested.
150 * @return Non-null list; empty when the employee has no recorded takeouts.
151 */
152 public synchronized List<Entry> getByEmployee(Long employeeId) {
153 return getAll().stream()
154 .filter(e -> employeeId.equals(e.employeeId))
155 .toList();
156 }
157
158 // =========================================================

160 // =========================================================
161
162 /**
163 * Prepends a new takeout entry to the log file.
164 *
165 * Prepending (index 0) keeps the file ordered newest-first without sorting on every write.
166 *
167 * @param employeeId ID of the employee who took the resource.
168 * @param employeeName Display name of the employee.
169 * @param employeeRole Role class name (e.g., {@code "Technician"}).
170 * @param resourceId Catalogue ID of the resource that was taken.
171 * @param resourceName Display name of the resource.
172 * @param storageId ID of the storage facility the resource was taken from.
173 * @param quantity Number of units taken.
174 * @throws RuntimeException When the updated list cannot be written to disk.
175 */
176 public synchronized void log(Long employeeId, String employeeName, String employeeRole,
177 Long resourceId, String resourceName, Long storageId, int quantity) {
178 List<Entry> all = getAll();
179 all.add(0, new Entry(employeeId, employeeName, employeeRole,
180 resourceId, resourceName, storageId, quantity));
181 write(all);
182 }
183
184 /**
185 * Serialises the entry list to the backing JSON file with pretty-printing.
186 *
187 * @param entries The full list to persist.
188 * @throws RuntimeException When the file cannot be written.
189 */
190 private void write(List<Entry> entries) {
191 try {
192 mapper.writerWithDefaultPrettyPrinter().writeValue(new File(path), entries);
193 } catch (IOException e) {
194 throw new RuntimeException("Error writing resource-take-logs.json", e);
195 }
196 }
