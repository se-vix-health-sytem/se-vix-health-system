84 /**
85 * GET /secretary/dashboard : render the secretary's overview dashboard.
86 *
87 * Resolves the secretary's department name by reloading a fresh entity from
88 * the database (avoiding lazy-loading failures on the detached session entity).
89 * Populates room and appointment summary counts.
90 *
91 * @param session HTTP session carrying the {@code "user"} Employee attribute,
92 * used to look up the secretary's assigned department name.
93 * @param model Receives {@code departmentName}, {@code totalRooms},
94 * {@code availableRooms}, {@code totalAppointments},
95 * and {@code totalAvailableBeds} attributes.
96 * @return Thymeleaf template {@code secretary/dashboard}.
97 */
98 @GetMapping("/dashboard")
99 public String dashboard(HttpSession session, Model model) {
100 int totalRooms = 0, availableRoomsCount = 0, totalAppointments = 0, totalBeds = 0;
101 try { totalRooms = roomService.getAllInpatientRooms().size(); } catch (Exception ignored) {}
102 try { availableRoomsCount = roomService.getAvailableRooms().size(); } catch (Exception ignored) {}
103 try { totalAppointments = appointmentRepository.findAll().size(); } catch (Exception ignored) {}
104 try { totalBeds = roomService.getTotalAvailableBeds(); } catch (Exception ignored) {}
105
106 String departmentName = "No department assigned";
107 Employee sessionUser = (Employee) session.getAttribute("user");
108 if (sessionUser != null) {
109 try {
110 Employee freshUser = employeeService.findById(sessionUser.getId());
111 if (freshUser != null && freshUser.getDepartment() != null) {
112 departmentName = freshUser.getDepartment().getName();
113 }
114 } catch (Exception ignored) {}
115 }
116
117 model.addAttribute("pageTitle", "Secretary Dashboard");
118 model.addAttribute("currentPage", "dashboard");
119 model.addAttribute("departmentName", departmentName);
120 model.addAttribute("totalRooms", totalRooms);
121 model.addAttribute("availableRooms", availableRoomsCount);
122 model.addAttribute("totalAppointments", totalAppointments);
123 model.addAttribute("totalAvailableBeds", totalBeds);
124 return "secretary/dashboard";
125 }

173 /**
174 * POST /secretary/patients/admit : admit a patient to a specific room.
175 *
176 * Delegates to {@link RoomService#admitPatient} via the Secretary domain object
177 * (resolved from session) and writes an audit entry.
178 *
179 * @param patientId Database ID of the patient to admit.
180 * @param roomId Database ID of the target inpatient room.
181 * @param session HTTP session; used to resolve the Secretary domain object
182 * so the business rule check inside the model runs correctly.
183 * @param redirectAttributes Flash attributes for the redirect.
184 * @return Redirect to {@code /secretary/rooms}.
185 */
186 @PostMapping("/patients/admit")
187 public String admitPatient(@RequestParam Long patientId,
188 @RequestParam Long roomId,
189 @RequestParam(required = false) Boolean confirmTransfer,
190 HttpSession session,
191 RedirectAttributes redirectAttributes) {
192 try {
193 Secretary secretary = getSecretaryFromSession(session);
194
195 // Check if the patient is already assigned to a room
196 InternationRoom currentRoom = roomService.findPatientRoom(patientId);
197
198 if (currentRoom != null && !Long.valueOf(currentRoom.getId()).equals(roomId)) {
199 if (!Boolean.TRUE.equals(confirmTransfer)) {
200 // Ask for confirmation before transferring
201 Patient patient = patientService.findById(patientId);
202 Room targetRoom = roomService.findById(roomId);
203 String targetNum = targetRoom.getNumber();
204 redirectAttributes.addFlashAttribute("transferWarning",
205 "Patient " + patient.getName() + " " + patient.getSurname()
206 + " is currently assigned to Room " + currentRoom.getNumber()
207 + ". Do you want to confirm transfer to Room " + targetNum + "?");
208 redirectAttributes.addFlashAttribute("pendingPatientId", patientId);
209 redirectAttributes.addFlashAttribute("pendingRoomId", roomId);
210 return "redirect:/secretary/rooms";
211 }
212 // Confirmed : dismiss from current room first, then admit to new room
213 roomService.dismissPatient(secretary, patientId, currentRoom.getId());
214 auditService.log("TRANSFER_PATIENT", "Patient", String.valueOf(patientId),
215 "Patient #" + patientId + " transferred from room #" + currentRoom.getId() + " to room #" + roomId);
216 }
217
218 roomService.admitPatient(secretary, patientId, roomId);
219 auditService.log("ADMIT_PATIENT", "Patient", String.valueOf(patientId),
220 "Secretary admitted patient #" + patientId + " to room #" + roomId);
221 redirectAttributes.addFlashAttribute("message",
222 "✅ Patient admitted to Room " + roomId + " successfully!");
223 } catch (Exception e) {
224 redirectAttributes.addFlashAttribute("error", "❌ Admission failed: " + e.getMessage());
225 }
226 return "redirect:/secretary/rooms";
227 }

491 /**
492 * Resolve and reload the authenticated Secretary from the HTTP session.
493 *
494 * The session {@code "user"} attribute is used because the Spring Security
495 * principal carries only the role string; the full Secretary domain object is
496 * needed by room and appointment service methods that enforce business rules.
497 * A fresh entity is reloaded from the database to avoid Hibernate detached-
498 * object issues.
499 *
500 * @param session HTTP session carrying the {@code "user"} attribute.
501 * @return A fully initialised {@link Secretary} from the database.
502 * @throws RuntimeException When the session user is absent or not a Secretary.
503 */
504 private Secretary getSecretaryFromSession(HttpSession session) {
505 Employee user = (Employee) session.getAttribute("user");
506 if (user instanceof Secretary secretary) {
507 // Reload from DB to get a fully initialized entity
508 Employee fresh = employeeService.findById(secretary.getId());
509 if (fresh instanceof Secretary s) {
510 return s;
511 }
512 }
513 throw new RuntimeException("Authenticated user is not a secretary");
514 }
