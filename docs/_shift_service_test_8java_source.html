33 /**
34 * Runs before every test.
35 *
36 * Creates a mocked AuditService,
37 * backs up the original JSON file,
38 * and initializes an empty shifts.json
39 * so tests remain independent.
40 */
41 @BeforeEach
42 void setUp() throws Exception {
43 auditService = mock(AuditService.class);
44
45 File file = new File(FILE_PATH);
46 file.getParentFile().mkdirs();
47
48 fileExistedBeforeTest = file.exists();
49
50 if (fileExistedBeforeTest) {
51 originalFileContent = Files.readString(file.toPath());
52 } else {
53 originalFileContent = null;
54 }
55
56 Files.writeString(file.toPath(), "[]");
57
58 shiftService = new ShiftService(auditService);
59 }
60
61 /**
62 * Runs after every test.
63 *
64 * Restores the original content of
65 * shifts.json so test execution does not
66 * modify application data permanently.
67 */
68 @AfterEach
69 void tearDown() throws Exception {
70 File file = new File(FILE_PATH);
71
72 if (fileExistedBeforeTest) {
73 Files.writeString(file.toPath(), originalFileContent);
74 } else {
75 Files.deleteIfExists(file.toPath());
76 }
77 }

79 /**
80 * Removes the generated shiftTypeName field
81 * from the JSON file so Jackson can deserialize
82 * Shift objects correctly during tests.
83 */
84 private void removeShiftTypeNameFromJson() throws Exception {
85 File file = new File(FILE_PATH);
86
87 String content = Files.readString(file.toPath());
88
89 content = content.replaceAll(
90 ",\\s*\"shiftTypeName\"\\s*:\\s*\"[^\"]*\"",
91 ""
92 );
93
94 Files.writeString(file.toPath(), content);
95 }
