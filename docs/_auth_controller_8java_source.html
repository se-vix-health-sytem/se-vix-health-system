22/**

83 /**
84 * POST /authenticate — verify Firebase credentials and open a staff session.
85 *
86 * Calls Firebase to exchange email/password for a UID, looks up the matching
87 * Employee, registers a Spring Security context in the HTTP session, then
88 * redirects to the role-specific dashboard. Returns to the login page with
89 * an error attribute on any failure.
90 *
91 * The session attribute {@code "user"} stores the Employee entity instead of
92 * using the Spring Security principal because employee domain objects carry
93 * additional context (department, specialty, etc.) that the principal does not.
94 *
95 * @param username Employee's Firebase email address.
96 * @param password Firebase account password (plain text over HTTPS).
97 * @param session HTTP session; receives {@code "user"} and {@code "role"} attributes.
98 * @param request Raw servlet request (passed to session-security plumbing).
99 * @param response Raw servlet response (passed to session-security plumbing).
100 * @param model Receives {@code "error"} when authentication fails.
101 * @return Redirect to the role-specific dashboard, or the {@code login}
102 * template with an error message.
103 */
104 @PostMapping("/authenticate")
105 public String authenticate(@RequestParam String username,
106 @RequestParam String password,
107 HttpSession session,
108 HttpServletRequest request,
109 HttpServletResponse response,
110 Model model) {
111
112 try {
113 String firebaseUid =
114 firebaseAuthService.signInWithEmailAndPassword(username, password);
115
116 Employee employee =
117 employeeService.findByFirebaseUid(firebaseUid);
118
119 if (employee == null) {
120 model.addAttribute("error", "Employee account not linked to Firebase");
121 return "login";
122 }
123
124 String role = employee.getClass().getSimpleName().toUpperCase();
125 session.setAttribute("user", employee);
126 session.setAttribute("role", role);
127
128 // Register with Spring Security so route-level protection works
129 SecurityContext ctx = SecurityContextHolder.createEmptyContext();
130 ctx.setAuthentication(new UsernamePasswordAuthenticationToken(
131 username, null, List.of(new SimpleGrantedAuthority("ROLE_" + role))
132 ));
133 SecurityContextHolder.setContext(ctx);
134 session.setAttribute(HttpSessionSecurityContextRepository.SPRING_SECURITY_CONTEXT_KEY, ctx);
135
136 if (employee instanceof com.nvivx.vixhealthsystem.model.person.employee.MedicalSpecialist) {
137 return "redirect:/medical-specialist/dashboard";
138 } else if (employee instanceof com.nvivx.vixhealthsystem.model.person.employee.Secretary) {
139 return "redirect:/secretary/dashboard";
140 } else if (employee instanceof com.nvivx.vixhealthsystem.model.person.employee.Technician) {
141 return "redirect:/technician/dashboard";
142 } else if (employee instanceof com.nvivx.vixhealthsystem.model.person.employee.Buyer) {
143 return "redirect:/buyer/dashboard";
144 } else if (employee instanceof com.nvivx.vixhealthsystem.model.person.employee.StaffManager) {
145 return "redirect:/staff-manager/dashboard";
146 }
147
148 } catch (Exception e) {
149 model.addAttribute("error", "Invalid Firebase credentials");
150 return "login";
151 }
152
153 model.addAttribute("error", "Invalid role");
154 return "login";
155 }

177 /**
178 * POST /select-role — dev-only role selector that bypasses Firebase.
179 *
180 * Looks up the first employee of the requested type from the database,
181 * seeds the HTTP session, registers a Spring Security context, then
182 * redirects to the matching dashboard. Not intended for production use.
183 *
184 * @param role One of: {@code MEDICAL_SPECIALIST}, {@code TECHNICIAN},
185 * {@code STAFF_MANAGER}, {@code SECRETARY}, {@code BUYER}.
186 * @param session HTTP session; receives {@code "user"} and {@code "role"} attributes.
187 * @return Redirect to the role-specific dashboard, or {@code redirect:/login}
188 * when the role string is unrecognized.
189 */
190 @PostMapping("/select-role")
191 public String processLogin(@RequestParam String role, HttpSession session) {
192 Class<? extends Employee> employeeType = switch (role) {
193 case "MEDICAL_SPECIALIST" -> com.nvivx.vixhealthsystem.model.person.employee.MedicalSpecialist.class;
194 case "TECHNICIAN" -> com.nvivx.vixhealthsystem.model.person.employee.Technician.class;
195 case "STAFF_MANAGER" -> com.nvivx.vixhealthsystem.model.person.employee.StaffManager.class;
196 case "SECRETARY" -> com.nvivx.vixhealthsystem.model.person.employee.Secretary.class;
197 case "BUYER" -> com.nvivx.vixhealthsystem.model.person.employee.Buyer.class;
198 default -> null;
199 };
200 if (employeeType == null) return "redirect:/login";
201
202 java.util.List<? extends Employee> matches = employeeService.findByRole(employeeType);
203 Employee demoUser = matches.isEmpty() ? null : matches.get(0);
204
205 String springRole = "ROLE_" + role.replace("_", "");
206 SecurityContext ctx = SecurityContextHolder.createEmptyContext();
207 ctx.setAuthentication(new UsernamePasswordAuthenticationToken(
208 role, null, List.of(new SimpleGrantedAuthority(springRole))
209 ));
210 SecurityContextHolder.setContext(ctx);
211 session.setAttribute(HttpSessionSecurityContextRepository.SPRING_SECURITY_CONTEXT_KEY, ctx);
212
213 if (demoUser != null) {
214 session.setAttribute("user", demoUser);
215 session.setAttribute("role", demoUser.getClass().getSimpleName().toUpperCase());
216 }
217
218 return switch (role) {
219 case "MEDICAL_SPECIALIST" -> "redirect:/medical-specialist/dashboard";
220 case "TECHNICIAN" -> "redirect:/technician/dashboard";
221 case "STAFF_MANAGER" -> "redirect:/staff-manager/dashboard";
222 case "SECRETARY" -> "redirect:/secretary/dashboard";
223 case "BUYER" -> "redirect:/buyer/dashboard";
224 default -> "redirect:/login";
225 };
226 }
