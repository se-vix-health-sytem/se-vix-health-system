14/**
15 * Manages employee vacation requests from creation through approval/denial (UC29, FR6.2).
16 *
17 * Requests are stored in {@code vacations.json} via {@link JsonVacationRepository} rather
18 * than the relational database, keeping vacation data separate from the core HR schema.
19 * The staff manager is the only actor who can add, approve, deny, or delete requests —
20 * employees themselves cannot self-serve vacation through this system.

22 * @see ShiftService

55 /**
56 * Creates a new vacation request with {@link VacationStatus#PENDING} status (UC29).
57 *
58 * The staff manager calls this on behalf of the employee; the request then waits for
59 * approval or denial.
60 *
61 * @param employeeId Database ID of the requesting employee.
62 * @param employeeName Display name used in audit log entries.
63 * @param startDate First day of the requested vacation period (inclusive).
64 * @param endDate Last day of the requested vacation period (inclusive).
65 * @param reason Free-text reason provided by the employee.
66 * @return The saved {@link VacationRequest} with a generated ID.
67 */
68 public VacationRequest addVacationRequest(int employeeId, String employeeName,
69 LocalDate startDate, LocalDate endDate,
70 String reason) {
71 VacationRequest request = new VacationRequest(
72 0, // ID assigned by repository
73 employeeId,
74 employeeName,
75 startDate,
76 endDate,
77 reason,
78 VacationStatus.PENDING
79 );
80 VacationRequest saved = repository.save(request);
81
82 auditService.log("CREATE_VACATION_REQUEST", "VacationRequest", String.valueOf(saved.getId()),
83 "Created vacation request for employee: " + employeeName + " (" + startDate + " to " + endDate + ")");
84
85 return saved;
86 }

88 /**
89 * Transitions a vacation request to {@link VacationStatus#APPROVED} (UC29).
90 *
91 * @param id ID of the request to approve.
92 * @return The updated {@link VacationRequest}.
93 * @throws com.nvivx.vixhealthsystem.exception.VacationNotFoundException When the ID does not exist.
94 */
95 public VacationRequest approveVacation(int id) {
96 VacationRequest request = repository.findById(id)
97 .orElseThrow(() -> new VacationNotFoundException("Vacation request not found: " + id));
98 request.setStatus(VacationStatus.APPROVED);
99 VacationRequest saved = repository.save(request);
100
101 auditService.log("APPROVE_VACATION", "VacationRequest", String.valueOf(id),
102 "Approved vacation for employee: " + request.getEmployeeName());
103
104 return saved;
105 }
106
107 /**
108 * Transitions a vacation request to {@link VacationStatus#DENIED} (UC29).
109 *
110 * @param id ID of the request to deny.
111 * @return The updated {@link VacationRequest}.
112 * @throws com.nvivx.vixhealthsystem.exception.VacationNotFoundException When the ID does not exist.

114 public VacationRequest denyVacation(int id) {
115 VacationRequest request = repository.findById(id)

123 return saved;

166 /**
167 * Checks whether an employee has any PENDING or APPROVED request that
168 * overlaps the given date range.
169 *
170 * Used before creating a new request to warn the staff manager about double-booking.
171 *
172 * @param employeeId Database ID of the employee.
173 * @param startDate Start of the proposed period (inclusive).
174 * @param endDate End of the proposed period (inclusive).
175 * @return {@code true} if at least one overlapping request exists.
176 */
177 public boolean hasOverlappingVacation(int employeeId, LocalDate startDate, LocalDate endDate) {
178 return getRequestsForEmployee(employeeId).stream()
179 .filter(v -> v.getStatus() == VacationStatus.APPROVED || v.getStatus() == VacationStatus.PENDING)
180 .anyMatch(v ->
181 !(v.getEndDate().isBefore(startDate) || v.getStartDate().isAfter(endDate))
182 );
183 }
184
185 /**
186 * Counts the total approved vacation days taken by an employee in a calendar year.
187 *
188 * Only approved requests are counted; PENDING and DENIED requests are excluded.
189 * The year filter is based on the request's start date.
190 *
191 * @param employeeId Database ID of the employee.
192 * @param year Calendar year to aggregate (e.g., {@code 2025}).
193 * @return Total number of approved days in that year.
194 */
195 public long getTotalVacationDaysInYear(int employeeId, int year) {
196 return getRequestsForEmployee(employeeId).stream()
197 .filter(v -> v.getStatus() == VacationStatus.APPROVED)
198 .filter(v -> v.getStartDate().getYear() == year)
199 .mapToLong(VacationRequest::getDaysRequested)
200 .sum();
201 }

219 }
