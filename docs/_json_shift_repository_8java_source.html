40 /**
41 * Returns all shifts stored in the JSON file.
42 *
43 * Returns an empty list when the file does not exist or cannot be parsed,
44 * rather than propagating an exception.
45 *
46 * @return mutable list of all shifts; never {@code null}
47 */
48 public List<Shift> findAll() {
49 try {
50 if (!file.exists()) return new ArrayList<>();
51
52 return mapper.readValue(file, new TypeReference<List<Shift>>() {});
53 } catch (Exception e) {
54 return new ArrayList<>();
55 }
56 }
57
58 /**
59 * Overwrites the JSON file with the complete shift list.
60 *
61 * @param shifts replacement list; must not be {@code null}
62 * @throws RuntimeException if the file cannot be written
63 */
64 public void saveAll(List<Shift> shifts) {
65 try {
66 mapper.writerWithDefaultPrettyPrinter().writeValue(file, shifts);
67 } catch (Exception e) {
68 throw new RuntimeException(e);
69 }
70 }
