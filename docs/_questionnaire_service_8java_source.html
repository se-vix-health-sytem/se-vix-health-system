8/**

10 *
11 * Given a symptom area selected by a site visitor, this service returns a recommended
12 * specialist type and an urgency level. The mapping is hardcoded rather than database-driven
13 * because the questionnaire content is owned by the medical team and changes infrequently.

15 * @see com.nvivx.vixhealthsystem.controllers.site.QuestionnaireController

24 /**
25 * Maps a symptom area to a specialist recommendation and urgency flag.
26 *
27 * @param symptomArea Key identifying the selected symptom (e.g., {@code "chest_pain"}).
28 * @return Map with keys: {@code specialist} (String), {@code message} (String),
29 * {@code urgency} (String).
30 */
31 public Map<String, Object> analyzeSymptoms(String symptomArea) {
32 Map<String, Object> result = new HashMap<>();
33
34 Map<String, String> symptomToSpecialist = getStringStringMap();
35
36 String specialist = symptomToSpecialist.getOrDefault(symptomArea, "General Medicine");
37 result.put("specialist", specialist);
38 result.put("message", "Based on your symptoms, we recommend consulting a " + specialist + " specialist.");
39 result.put("urgency", determineUrgency(symptomArea));
40
41 return result;
42 }

48 /** Returns the static symptom-to-specialist lookup table. */
49 private static @NonNull Map<String, String> getStringStringMap() {
50 Map<String, String> symptomToSpecialist = new HashMap<>();
51 symptomToSpecialist.put("chest_pain", "Cardiology");
52 symptomToSpecialist.put("shortness_breath", "Cardiology");
53 symptomToSpecialist.put("headache", "Neurology");
54 symptomToSpecialist.put("joint_pain", "Orthopaedics");
55 symptomToSpecialist.put("abdominal_pain", "Gynaecology/General Medicine");
56 symptomToSpecialist.put("fever", "General Medicine");
57 symptomToSpecialist.put("skin_rash", "Dermatology");
58 return symptomToSpecialist;
59 }
60
61 /**
62 * Returns a human-readable urgency string for the given symptom.
63 *
64 * Symptoms not in the urgency table default to MEDIUM priority.
65 *
66 * @param symptom Symptom key.
67 * @return Urgency label (e.g., {@code "HIGH - Seek immediate attention"}).
68 */
69 private String determineUrgency(String symptom) {
70 Map<String, String> urgency = new HashMap<>();
71 urgency.put("chest_pain", "HIGH - Seek immediate attention");
72 urgency.put("shortness_breath", "HIGH - Seek immediate attention");
73 urgency.put("headache", "MEDIUM - Schedule an appointment soon");
74 urgency.put("joint_pain", "LOW - Can schedule regular appointment");
75 return urgency.getOrDefault(symptom, "MEDIUM - Consultation recommended");
76 }
