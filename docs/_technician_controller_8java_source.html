135 /**
136 * POST /technician/machines/update-status — change the operational status of a machine.
137 *
138 * @param machineId Database ID of the machine to update.
139 * @param status String representation of a {@link MachineStatus} enum value
140 * (e.g., {@code "WORKING"}, {@code "FAULTY"}, {@code "MAINTENANCE"}).
141 * @param model Receives a success or error {@code message} attribute.
142 * @return Thymeleaf template {@code technician/result}.
143 */
144 @PostMapping("/machines/update-status")
145 public String updateMachineStatus(@RequestParam Long machineId,
146 @RequestParam String status,
147 Model model) {
148 try {
149 MachineStatus newStatus = MachineStatus.valueOf(status);
150 var machine = machineryService.updateMachineStatus(machineId, newStatus);
151
152 String statusEmoji = newStatus == MachineStatus.WORKING ? "🟢" :
153 newStatus == MachineStatus.FAULTY ? "🔴" : "🟡";
154
155 String alertMsg = newStatus == MachineStatus.FAULTY ?
156 "\n\n⚠️ ALERT: Maintenance team has been notified!" : "";
157
158 model.addAttribute("pageTitle", "Machine Status Updated");
159 model.addAttribute("message",
160 "✅ Machine Status Updated!\n\n" +
161 "Machine: " + machine.getName() + " (ID: " + machineId + ")\n" +
162 "New Status: " + statusEmoji + " " + newStatus + alertMsg);
163 } catch (Exception e) {
164 model.addAttribute("pageTitle", "Error");
165 model.addAttribute("message", "❌ Error: " + e.getMessage());
166 }
167 return "technician/result";
168 }

248 /**
249 * Resolve and reload the authenticated Technician from the HTTP session.
250 *
251 * The session {@code "user"} attribute is used because {@link MachineryService}
252 * methods require the full Technician domain object. Falls back to a transient
253 * {@code new Technician()} when the session carries no Technician.
254 *
255 * @param session HTTP session carrying the {@code "user"} attribute.
256 * @return A fully initialised {@link Technician} from the database, or a
257 * transient instance as a fallback.
258 */
259 private Technician getTechnicianFromSession(HttpSession session) {
260 Employee user = (Employee) session.getAttribute("user");
261 if (user instanceof Technician t) {
262 Employee fresh = employeeService.findById(t.getId());
263 if (fresh instanceof Technician tech) {
264 return tech;
265 }
266 }
267 return new Technician();
268 }
