8/**
9 * @brief Simulates SPID and CIE government-identity login pages for demo purposes.
10 *
11 * In a production deployment these endpoints would be replaced by redirects to
12 * the real Italian government identity providers (SPID and CIE). Here they
13 * host simplified mock login pages that accept a patient's fiscal code as the
14 * "assertion token", perform minimal validation, and then hand off to the real
15 * patient auth callback in {@link PatientAuthController}.
16 *
17 * Auth flow:
18 * {@code /patient/login} → {@code /mock-spid/login} → POST {@code /mock-spid/authenticate}
19 * → redirect {@code /patient/spid-callback?fiscalCode=...}
20 * → {@code /mock-cie/login} → POST {@code /mock-cie/authenticate}
21 * → redirect {@code /patient/cie-callback?fiscalCode=...}
22 *
23 * These endpoints are public (no authentication required) because they are the
24 * entry point to the login flow.
25 *
26 * @see PatientAuthController
27 * @see PatientService
28 */
29@Controller
30public class MockSpidController {

52 /**
53 * POST /mock-spid/authenticate — validate the fiscal code and PIN, then redirect to the SPID callback.
54 *
55 * Checks that the fiscal code is exactly 16 characters, the PIN is at least
56 * 6 digits long, and the patient exists in the database. On success the
57 * fiscal code is passed as a query parameter to the patient auth callback,
58 * where it acts as the identity assertion token.
59 *
60 * @param fiscalCode Italian fiscal code (codice fiscale); trimmed and uppercased.
61 * @param pin Numeric PIN; must be at least 6 digits.
62 * @param model Receives an {@code error} attribute on validation failure.
63 * @return Redirect to {@code /patient/spid-callback?fiscalCode=...} on success,
64 * or the {@code mock-spid/login} template with an error message.
65 */
66 @PostMapping("/mock-spid/authenticate")
67 public String spidAuthenticate(@RequestParam String fiscalCode,
68 @RequestParam String pin,
69 Model model) {
70 String fc = fiscalCode.trim().toUpperCase();
71
72 if (fc.length() != 16) {
73 model.addAttribute("error", "Codice fiscale non valido. Deve essere di 16 caratteri.");
74 return "mock-spid/login";
75 }
76 if (pin == null || pin.isBlank() || pin.length() < 6) {
77 model.addAttribute("error", "PIN non valido. Inserisci almeno 6 cifre.");
78 return "mock-spid/login";
79 }
80
81 if (patientService.findByFiscalCode(fc).isEmpty()) {
82 model.addAttribute("error", "Nessun account trovato per il codice fiscale inserito.");
83 return "mock-spid/login";
84 }
85
86 // Hand off to the patient auth callback; fiscalCode acts as the "token"
87 return "redirect:/patient/spid-callback?fiscalCode=" + fc;
88 }
