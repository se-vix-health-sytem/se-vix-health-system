45 /**
46 * Initialises the Firebase Admin SDK using the service-account JSON.
47 *
48 * Is a no-op when:
49 * - {@code firebase.enabled=false}, or
50 * - Firebase is already initialised (avoids duplicate-app errors on hot-reload).
51 *
52 * @throws Exception When the credential file cannot be read or parsed.
53 */
54 @PostConstruct
55 public void initializeFirebase() throws Exception {
56
57 if (!enabled) {
58 return;
59 }
60
61 if (!FirebaseApp.getApps().isEmpty()) {
62 return;
63 }
64
65 try (InputStream inputStream = serviceAccount.getInputStream()) {
66 FirebaseOptions options = FirebaseOptions.builder()
67 .setCredentials(GoogleCredentials.fromStream(inputStream))
68 .build();
69
70 FirebaseApp.initializeApp(options);
71 }
72 }
