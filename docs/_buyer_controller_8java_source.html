98 /**
99 * GET /buyer/inventory — list all resources currently in storage.
100 *
101 * Aggregates stock quantities across all storage locations and wraps each
102 * entry in an {@link InventoryItem} for the view. Resources with fewer than
103 * 50 units are flagged as low-stock. An empty list is substituted and an
104 * error banner is shown if the service call fails.
105 *
106 * @param model Receives {@code resources} (sorted by name), {@code pageTitle},
107 * and {@code isLowStockView} attributes.
108 * @param session HTTP session (retained for future session-based filtering).
109 * @return Thymeleaf template {@code buyer/inventory}.
110 */
111 @GetMapping("/inventory")
112 public String viewInventory(Model model, HttpSession session) {
113 try {
114 var totalInventory = inventoryService.getTotalInventory();
115
116 var resourceList = totalInventory.entrySet().stream()
117 .map(entry -> new InventoryItem(
118 entry.getKey().getId(),
119 entry.getKey().getName(),
120 entry.getKey().getDescription(),
121 entry.getValue(),
122 entry.getValue() < 50,
123 entry.getKey().getPrice(),
124 "units"
125 ))
126 .sorted((a, b) -> a.getName().compareToIgnoreCase(b.getName()))
127 .collect(Collectors.toList());
128
129 model.addAttribute("resources", resourceList);
130 model.addAttribute("pageTitle", "Inventory Management");
131 model.addAttribute("currentPage", "inventory");
132 model.addAttribute("isLowStockView", false);
133 } catch (Exception e) {
134 model.addAttribute("resources", new ArrayList<>());
135 model.addAttribute("error", "Could not load inventory: " + e.getMessage());
136 }
137 return "buyer/inventory";
138 }

182 /**
183 * POST /buyer/inventory/add — create a new resource and add initial stock.
184 *
185 * Creates the resource entity via {@link InventoryService}, then calls the
186 * domain method {@code Buyer.addResourceToStorage} to record the initial
187 * quantity. The acting buyer is resolved from the session.
188 *
189 * @param name Human-readable resource name.
190 * @param description Short description of the resource.
191 * @param quantity Initial quantity to add to storage; must be positive.
192 * @param price Unit price in euros.
193 * @param unit Unit of measure label (e.g., "units", "boxes").
194 * @param session HTTP session; used to resolve the authenticated Buyer via
195 * {@code "user"} — needed because the Buyer domain object
196 * contains the business logic for adding stock.
197 * @param model Receives a success or error {@code message} attribute.
198 * @return Thymeleaf template {@code buyer/result}.
199 */
200 @PostMapping("/inventory/add")
201 public String addResource(@RequestParam String name,
202 @RequestParam String description,
203 @RequestParam int quantity,
204 @RequestParam float price,
205 @RequestParam String unit,
206 HttpSession session,
207 Model model) {
208 try {
209 Buyer buyer = getBuyerFromSession(session);
210 BigDecimal bigDecimalPrice = BigDecimal.valueOf(price);
211 var newResource = inventoryService.createResource(name, description, bigDecimalPrice);
212
213 // Domain: buyer adds resource to storage via model method
214 inventoryService.addResourceToStorage(buyer, newResource.getId(), quantity);
215
216 model.addAttribute("pageTitle", "Resource Added Successfully");
217 model.addAttribute("message",
218 "✅ Resource Added!\n\n" +
219 "Name: " + name + "\n" +
220 "Quantity: " + quantity + " " + unit + "\n" +
221 "Price per unit: €" + price + "\n\n" +
222 "Resource ID: " + newResource.getId());
223 } catch (Exception e) {
224 model.addAttribute("pageTitle", "Error");
225 model.addAttribute("message", "❌ Error: " + e.getMessage());
226 }
227 return "buyer/result";
228 }

384 /**
385 * Resolve and reload the authenticated Buyer from the HTTP session.
386 *
387 * The session attribute {@code "user"} is read because the Spring Security
388 * principal only carries the role string, not the full domain object needed
389 * by Buyer's storage mutation methods. A fresh entity is loaded from the
390 * database to avoid Hibernate detached-object issues.
391 *
392 * @param session HTTP session carrying the {@code "user"} attribute.
393 * @return A fully initialised {@link Buyer} from the database, or a
394 * transient {@code new Buyer()} when the session holds no Buyer.
395 */
396 private Buyer getBuyerFromSession(HttpSession session) {
397 Employee user = (Employee) session.getAttribute("user");
398 if (user instanceof Buyer b) {
399 Employee fresh = employeeService.findById(b.getId());
400 if (fresh instanceof Buyer buyer) {
401 return buyer;
402 }
403 }
404 return new Buyer();
405 }
