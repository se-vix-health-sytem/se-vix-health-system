18/**
19 * Manages employee shift assignments, backed by a JSON file rather than the database.
20 *
21 * Shifts are stored in {@code src/main/resources/storage/shifts.json}. The JSON
22 * approach was chosen because shift data changes frequently and has no relational
23 * dependencies that require foreign-key constraints. Every write is flushed to disk
24 * immediately (no caching) so the file always reflects the current state.
25 *
26 * The ID sequence is rebuilt from the existing JSON on startup, which means IDs
27 * continue incrementing even after a server restart — they are never reused.
28 *
29 * All mutating operations are audit-logged via {@link AuditService}.
30 *
31 * @see VacationService
32 * @see com.nvivx.vixhealthsystem.model.staff.Shift
33 */
34@Service
35public class ShiftService {
36
37 // =========================================================
38 // FIELDS
39 // =========================================================
40
41 private final ObjectMapper mapper;
42 /** Path to the JSON shift store, relative to the working directory. */
43 private final String shiftsPath = "src/main/resources/storage/shifts.json";
44 private final AuditService auditService;
45 /** Auto-increments on every new shift; seeded from the max existing ID at startup. */
46 private final AtomicLong idGenerator = new AtomicLong(1);
47
48 // =========================================================
49 // CONSTRUCTOR
50 // =========================================================
51
52 /**
53 * Wires up dependencies and ensures the JSON file exists.
54 *
55 * @param auditService Used to record shift assignments and removals.
56 */
57 public ShiftService(AuditService auditService) {
58 this.auditService = auditService;
59 this.mapper = new ObjectMapper();
60 this.mapper.registerModule(new JavaTimeModule());
61 this.mapper.disable(com.fasterxml.jackson.databind.SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
62 initializeStorage();
63 }

69 /** Creates the JSON file if it doesn't exist, then seeds the ID generator. */
70 private void initializeStorage() {
71 File file = new File(shiftsPath);
72 if (!file.exists()) {
73 try {
74 file.getParentFile().mkdirs();
75 mapper.writeValue(file, new ArrayList<Shift>());
76 } catch (IOException e) {
77 throw new RuntimeException("Could not create shifts.json", e);
78 }
79 }
80
81 // Initialize ID generator based on existing shifts
82 List<Shift> existing = findAll();
83 long maxId = existing.stream()
84 .mapToLong(Shift::getId)
85 .max()
86 .orElse(0L);
87 idGenerator.set(maxId + 1);
88 }
89
90 /** Reads the entire JSON file and deserialises it into a list of shifts. */
91 private List<Shift> findAll() {
92 try {
93 File file = new File(shiftsPath);
94 if (!file.exists() || file.length() == 0) {
95 return new ArrayList<>();
96 }
97 return mapper.readValue(file, new TypeReference<List<Shift>>() {});
98 } catch (IOException e) {
99 throw new RuntimeException("Error reading shifts.json", e);
100 }
101 }
102
103 /** Serialises the full list back to disk, overwriting the file. */
104 private void saveAll(List<Shift> shifts) {
105 try {
106 mapper.writerWithDefaultPrettyPrinter().writeValue(new File(shiftsPath), shifts);
107 } catch (IOException e) {
108 throw new RuntimeException("Error writing shifts.json", e);
109 }
110 }

116 /**
117 * Returns all shifts assigned to a specific employee.
118 *
119 * @param employeeId The employee's primary key.
120 * @return List of shifts; empty if the employee has none.
121 */
122 public List<Shift> getShiftsForEmployee(Long employeeId) {
123 return findAll().stream()
124 .filter(s -> s.getEmployeeId().equals(employeeId))
125 .collect(Collectors.toList());
126 }
127
128 /**
129 * Returns all shifts that fall within the 7-day window starting on {@code startDate}.
130 *
131 * @param startDate First day of the week (inclusive).
132 * @return Shifts across all employees for those 7 days.
133 */
134 public List<Shift> getShiftsForWeek(LocalDate startDate) {
135 LocalDate endDate = startDate.plusDays(6);
136 return findAll().stream()
137 .filter(s -> !s.getDate().isBefore(startDate) && !s.getDate().isAfter(endDate))
138 .collect(Collectors.toList());
139 }
140
141 /**
142 * Returns an employee's shifts within an inclusive date range.
143 *
144 * @param employeeId The employee's primary key.
145 * @param startDate Start of the range (inclusive).
146 * @param endDate End of the range (inclusive).
147 * @return Matching shifts; empty list if none found.
148 */
149 public List<Shift> getShiftsForEmployeeBetweenDates(Long employeeId, LocalDate startDate, LocalDate endDate) {
150 return findAll().stream()
151 .filter(s -> s.getEmployeeId().equals(employeeId))
152 .filter(s -> !s.getDate().isBefore(startDate) && !s.getDate().isAfter(endDate))
153 .collect(Collectors.toList());
154 }
155
156 /**
157 * Checks whether an employee is scheduled on a given date.
158 *
159 * @param employeeId The employee's primary key.
160 * @param date The date to check.
161 * @return {@code true} if at least one shift exists for that day.
162 */
163 public boolean isEmployeeOnShift(Long employeeId, LocalDate date) {
164 return getShiftsForEmployee(employeeId).stream()
165 .anyMatch(s -> s.getDate().equals(date));
166 }

172 /**
173 * Assigns a shift to an employee on a given date, replacing any existing shift for that day.
174 *
175 * Only one shift per employee per day is allowed — if one already exists, it is silently
176 * removed before the new one is inserted.
177 *
178 * @param employeeId The employee's primary key.
179 * @param date The date of the shift.
180 * @param shiftType One of {@code "MORNING"}, {@code "AFTERNOON"}, or {@code "NIGHT"}.
181 */
182 public void assignShift(Long employeeId, LocalDate date, String shiftType) {
183 List<Shift> shifts = findAll();
184
185 // Remove existing shift for same employee on same day
186 shifts.removeIf(s -> s.getEmployeeId().equals(employeeId) && s.getDate().equals(date));
187
188 Shift newShift = new Shift();
189 newShift.setId(idGenerator.getAndIncrement());
190 newShift.setEmployeeId(employeeId);
191 newShift.setDate(date);
192 newShift.setShiftType(shiftType);
193 newShift.setNotes("Assigned by Staff Manager");
194
195 shifts.add(newShift);
196 saveAll(shifts);
197
198 auditService.log("ASSIGN_SHIFT", "Shift", String.valueOf(newShift.getId()),
199 "Assigned " + shiftType + " shift to employee " + employeeId + " on " + date);
200 }
201
202 /**
203 * Removes a shift by ID. Does nothing if the ID doesn't exist.
204 *
205 * @param shiftId ID of the shift to delete.
206 */
207 public void removeShift(Long shiftId) {
208 List<Shift> shifts = findAll();
209 Shift removed = shifts.stream()
210 .filter(s -> s.getId().equals(shiftId))
211 .findFirst()
212 .orElse(null);
213
214 shifts.removeIf(s -> s.getId().equals(shiftId));
215 saveAll(shifts);
216
217 if (removed != null) {
218 auditService.log("REMOVE_SHIFT", "Shift", String.valueOf(shiftId),
219 "Removed shift for employee " + removed.getEmployeeId() + " on " + removed.getDate());
220 }
221 }
222
223 /** @brief Returns all shifts in the system, used by the staff manager overview. */
224 public List<Shift> getAllShifts() {
225 return findAll();
226 }
