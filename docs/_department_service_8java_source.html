134 /**
135 * Builds a stable {@code specialistId → image-path} map for every specialist
136 * across all departments using only two database queries (all departments + all employees).
137 *
138 * Male and female counters reset per department; the image pool wraps at index 2
139 * to avoid {@code null} paths when there are more than two doctors of the same gender.
140 *
141 * @return {@link LinkedHashMap} ordered by department iteration order; never {@code null}.
142 */
143 public Map<Long, String> getAllDoctorImageMap() {
144 List<Department> departments = departmentRepository.findAll();
145 List<Employee> allEmployees = employeeRepository.findAll();
146
147 Map<Long, List<MedicalSpecialist>> byDept = allEmployees.stream()
148 .filter(e -> e instanceof MedicalSpecialist)
149 .map(e -> (MedicalSpecialist) e)
150 .filter(ms -> ms.getDepartment() != null)
151 .collect(Collectors.groupingBy(ms -> ms.getDepartment().getId()));
152
153 Map<Long, String> result = new LinkedHashMap<>();
154 for (Department dept : departments) {
155 List<MedicalSpecialist> docs = new ArrayList<>(byDept.getOrDefault(dept.getId(), List.of()));
156 docs.sort(Comparator.comparing(MedicalSpecialist::getId));
157 String deptKey = dept.getName().toLowerCase().replace(" ", "");
158 int maleCount = 0, femaleCount = 0;
159 for (MedicalSpecialist doc : docs) {
160 boolean isMale = doc.getGender() == 'M' || doc.getGender() == 'm';
161 String suffix = isMale
162 ? "m" + (++maleCount > 2 ? ((maleCount - 1) % 2) + 1 : maleCount)
163 : "f" + (++femaleCount > 2 ? ((femaleCount - 1) % 2) + 1 : femaleCount);
164 result.put(doc.getId(), "/images/doctors/" + deptKey + "_" + suffix + ".jpg");
165 }
166 }
167 return result;
168 }

170 /**
171 * Builds a stable {@code specialistId → image-path} map for a single department.
172 *
173 * Doctors are sorted by ID so that adding or removing staff never reshuffles
174 * existing photo assignments. The counter wraps at 2 per gender to stay within
175 * the available static asset set.
176 *
177 * @param departmentId ID of the department to map.
178 * @return Ordered map of specialist ID to image path; empty map when
179 * the department does not exist.
180 */
181 public Map<Long, String> getDoctorImageMap(Long departmentId) {
182 Department dept = getDepartmentById(departmentId);
183 if (dept == null) return Map.of();
184
185 List<MedicalSpecialist> doctors = getDoctorsByDepartment(departmentId);
186 doctors.sort(Comparator.comparing(d -> d.getId()));
187
188 String deptKey = dept.getName().toLowerCase().replace(" ", "");
189 Map<Long, String> imageMap = new LinkedHashMap<>();
190 int maleCount = 0, femaleCount = 0;
191 for (MedicalSpecialist doc : doctors) {
192 boolean isMale = doc.getGender() == 'M' || doc.getGender() == 'm';
193 String suffix = isMale
194 ? "m" + (++maleCount > 2 ? ((maleCount - 1) % 2) + 1 : maleCount)
195 : "f" + (++femaleCount > 2 ? ((femaleCount - 1) % 2) + 1 : femaleCount);
196 imageMap.put(doc.getId(), "/images/doctors/" + deptKey + "_" + suffix + ".jpg");
197 }
198 return imageMap;
199 }
