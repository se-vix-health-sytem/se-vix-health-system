13/**
14 * @brief Controller for the payment workflow : base URL {@code /payment}.
15 *
16 * Manages the end-to-end payment flow for appointment fees: showing the
17 * payment form, processing a real card payment via {@link PaymentService},
18 * querying transaction status, and a fake-pay shortcut for demo/test purposes
19 * that simulates a successful transaction without real card data. After
20 * a successful payment the appointment status is set to CONFIRMED.
21 *
22 * @see PaymentService
23 * @see JsonAppointmentRepository
24 */
25@Controller
26@RequestMapping("/payment")
27public class PaymentController {

42 /**
43 * GET /payment/appointment/{appointmentId} : render the payment form for an appointment.
44 *
45 * Looks up the appointment to confirm it exists before presenting the
46 * payment form; redirects back to the appointments list if it does not.
47 *
48 * @param appointmentId Numeric ID of the appointment to pay for.
49 * @param model Receives {@code appointmentId}, {@code amount}, and {@code pageTitle}.
50 * @return Thymeleaf template {@code payment/payment-form}, or
51 * redirect to {@code /patient/appointments?error=...} when
52 * the appointment is not found.
53 */
54 @GetMapping("/appointment/{appointmentId}")
55 public String showPaymentPage(@PathVariable int appointmentId, Model model) {
56 // Check if appointment exists
57 var appointment = appointmentRepository.findById(appointmentId);
58 if (appointment == null) {
59 return "redirect:/patient/appointments?error=Appointment+not+found";
60 }
61
62 model.addAttribute("appointmentId", appointmentId);
63 model.addAttribute("amount", paymentService.getAppointmentCost(appointmentId));
64 model.addAttribute("pageTitle", "Complete Payment");
65 return "payment/payment-form";
66 }
67

74 @GetMapping("/status/{paymentId}")

100 /**

103 * Accepts a full {@link PaymentRequest} (card number, CVV, expiry, amount)

118 /**
119 * POST /payment/fake-pay : simulate a successful payment for demo purposes.
120 *
121 * Builds a hard-coded {@link PaymentRequest} with a test card number and
122 * calls the real payment service path so that the same confirmation and
123 * status-update logic runs as in a live transaction. On success the
124 * appointment is marked as paid and CONFIRMED before redirecting.
125 *
126 * @param appointmentId Numeric ID of the appointment to mark as paid.
127 * @param amount Amount to simulate; shown in the success flash message.
128 * @param redirectAttributes Flash attributes carrying a success {@code message}
129 * or an {@code error} back to the confirmation page.
130 * @return Redirect to {@code /payment/confirmation/{appointmentId}}.
131 */
132 @PostMapping("/fake-pay")
133 public String fakePayment(@RequestParam int appointmentId,
134 @RequestParam float amount,
135 RedirectAttributes redirectAttributes) {
136 // Check if appointment exists
137 var appointment = appointmentRepository.findById(appointmentId);
138 if (appointment == null) {
139 redirectAttributes.addFlashAttribute("error", "Appointment not found");
140 return "redirect:/patient/appointments";
141 }
142
143 // Simple fake payment for demo
144 PaymentRequest request = new PaymentRequest();
145 request.setAppointmentId(appointmentId);
146 request.setAmount(amount);
147 request.setCardNumber("4111111111111111");
148 request.setCvv("123");
149 request.setExpiryDate("12/25");
150
151 PaymentResponse response = paymentService.processPayment(request);
152
153 if ("SUCCESS".equals(response.getStatus())) {
154 // Update appointment payment status and confirm it
155 appointment.pay();
156 appointmentRepository.save(appointment);
157
158 redirectAttributes.addFlashAttribute("message",
159 "✅ Payment of €" + amount + " successful! Transaction ID: " + response.getTransactionId());
160 } else {
161 redirectAttributes.addFlashAttribute("error", "Payment failed: " + response.getMessage());
162 }
163
164 return "redirect:/payment/confirmation/" + appointmentId;
165 }
