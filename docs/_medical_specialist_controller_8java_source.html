180 /**
181 * POST /medical-specialist/patients/{patientId}/add-surgery : schedule a surgery for a patient.
182 *
183 * The acting specialist is resolved from the session to invoke the domain
184 * method {@code MedicalSpecialist.scheduleSurgeryForPatient}. The session is
185 * consulted rather than the Spring Security principal because only the full
186 * MedicalSpecialist domain object carries the scheduling business logic.
187 * The operation is audited via {@link AuditService}.
188 *
189 * @param patientId Database ID of the patient receiving the surgery.
190 * @param surgeryName Name / type of the surgery.
191 * @param surgeryDescription Optional narrative description; defaults to empty string.
192 * @param surgeryDateTime ISO datetime string from an HTML {@code datetime-local} input
193 * ({@code yyyy-MM-ddTHH:mm} or {@code yyyy-MM-ddTHH:mm:ss}).
194 * @param roomId Optional ID of the {@link SpecializedRoom} to use.
195 * @param session HTTP session carrying the {@code "user"} Employee attribute.
196 * @param redirectAttributes Flash attributes for the redirect.
197 * @return Redirect to {@code /medical-specialist/patients/{patientId}/record}.
198 */
199 @PostMapping("/patients/{patientId}/add-surgery")
200 public String addSurgery(@PathVariable Long patientId,
201 @RequestParam String surgeryName,
202 @RequestParam(required = false) String surgeryDescription,
203 @RequestParam String surgeryDateTime,
204 @RequestParam(required = false) Long roomId,
205 HttpSession session,
206 RedirectAttributes redirectAttributes) {
207 try {
208 Patient patient = medicalRecordService.getPatientWithMedicalRecord(patientId);
209 if (patient.getMedicalRecord() == null) {
210 throw new RuntimeException("Patient has no medical record");
211 }
212
213 SpecializedRoom specializedRoom = null;
214 if (roomId != null) {
215 Room roomEntity = roomRepository.findById(roomId)
216 .orElseThrow(() -> new RuntimeException("Room not found: " + roomId));
217 if (!(roomEntity instanceof SpecializedRoom)) {
218 throw new RuntimeException("Selected room is not a specialized operating room");
219 }
220 specializedRoom = (SpecializedRoom) roomEntity;
221 }
222
223 Employee user = (Employee) session.getAttribute("user");
224 MedicalSpecialist specialist = null;
225 if (user != null) {
226 Employee fresh = employeeService.findById(user.getId());
227 if (fresh instanceof MedicalSpecialist ms) specialist = ms;
228 }
229
230 Surgery surgery = new Surgery();
231 surgery.setName(surgeryName);
232 surgery.setDescription(surgeryDescription != null ? surgeryDescription : "");
233 surgery.setDateTime(LocalDateTime.parse(surgeryDateTime, DT_FORM));
234 surgery.setSpecializedRoom(specializedRoom);
235
236 if (specialist != null) {
237 specialist.scheduleSurgeryForPatient(patient, surgery);
238 } else {
239 surgery.setMedicalRecord(patient.getMedicalRecord());
240 }
241
242 surgeryRepository.save(surgery);
243 auditService.log("ADD_SURGERY", "MedicalRecord",
244 String.valueOf(patient.getMedicalRecord().getId()),
245 "Scheduled surgery '" + surgeryName + "' for patient " + patientId);
246 redirectAttributes.addFlashAttribute("successMessage", "✅ Surgery scheduled successfully!");
247 } catch (Exception e) {
248 redirectAttributes.addFlashAttribute("errorMessage", "❌ Error scheduling surgery: " + e.getMessage());
249 }
250 return "redirect:/medical-specialist/patients/" + patientId + "/record";
251 }

379 /**
380 * GET /medical-specialist/calendar : display a monthly calendar with event markers.
381 *
382 * Collects shift dates, appointment dates, surgery dates, and approved vacation
383 * ranges for the requested month and passes them to the template as JSON-serialisable
384 * string lists so the front-end JavaScript grid can render markers without an
385 * additional API call.
386 *
387 * @param month Optional {@code yyyy-MM} string selecting the displayed month;
388 * defaults to the current month when absent or unparseable.
389 * @param session HTTP session carrying the {@code "user"} Employee attribute.
390 * @param model Receives {@code calYear}, {@code calMonth}, {@code monthLabel},
391 * {@code prevMonth}, {@code nextMonth}, {@code shiftDates},
392 * {@code apptDates}, {@code surgeryDates}, and {@code vacRanges}.
393 * @return Thymeleaf template {@code medical-specialist/calendar}, or
394 * {@code redirect:/login} when no session user is found.
395 */
396 @GetMapping("/calendar")
397 public String viewCalendar(@RequestParam(required = false) String month,
398 HttpSession session, Model model) {
399 Employee user = (Employee) session.getAttribute("user");
400 if (user == null) return "redirect:/login";
401
402 YearMonth ymTemp;
403 try {
404 ymTemp = (month != null && !month.isBlank()) ? YearMonth.parse(month) : YearMonth.now();
405 } catch (Exception e) {
406 ymTemp = YearMonth.now();
407 }
408 final YearMonth ym = ymTemp;
409
410 YearMonth prev = ym.minusMonths(1);
411 YearMonth next = ym.plusMonths(1);
412
413 // Collect event dates for the month : each service call wrapped to prevent a single failure from crashing the page
414 Set<LocalDate> shiftDates = new HashSet<>();
415 try {
416 shiftDates = shiftService.getShiftsForEmployee(user.getId()).stream()
417 .filter(s -> s.getDate() != null && YearMonth.from(s.getDate()).equals(ym))
418 .map(Shift::getDate).collect(Collectors.toSet());
419 } catch (Exception ignored) {}
420
421 Set<LocalDate> apptDates = new HashSet<>();
422 try {
423 apptDates = appointmentRepository.findAll().stream()
424 .filter(a -> a.getMedicalSpecialist() != null
425 && a.getMedicalSpecialist().getId().equals(user.getId())
426 && !"CANCELLED".equals(a.getStatus())
427 && a.getDateTime() != null
428 && YearMonth.from(a.getDateTime().toLocalDate()).equals(ym))
429 .map(a -> a.getDateTime().toLocalDate()).collect(Collectors.toSet());
430 } catch (Exception ignored) {}
431
432 Set<LocalDate> surgeryDates = new HashSet<>();
433 try {
434 surgeryDates = surgeryRepository.findByMedicalSpecialistId(user.getId()).stream()
435 .filter(s -> s.getDateTime() != null
436 && YearMonth.from(s.getDateTime().toLocalDate()).equals(ym))
437 .map(s -> s.getDateTime().toLocalDate()).collect(Collectors.toSet());
438 } catch (Exception ignored) {}
439
440 List<VacationRequest> vacations = new java.util.ArrayList<>();
441 try { vacations = vacationService.getApprovedRequestsForEmployee(user.getId().intValue()); } catch (Exception ignored) {}
442
443 // Pass date sets as simple String lists : JavaScript builds the grid
444 List<String> shiftDatesList = shiftDates.stream().map(LocalDate::toString).collect(Collectors.toList());
445 List<String> apptDatesList = apptDates.stream().map(LocalDate::toString).collect(Collectors.toList());
446 List<String> surgeryDatesList = surgeryDates.stream().map(LocalDate::toString).collect(Collectors.toList());
447 List<Map<String, String>> vacRanges = vacations.stream()
448 .filter(v -> v.getStartDate() != null && v.getEndDate() != null)
449 .map(v -> {
450 Map<String, String> m = new HashMap<>();
451 m.put("start", v.getStartDate().toString());
452 m.put("end", v.getEndDate().toString());
453 return m;
454 }).collect(Collectors.toList());
455
456 model.addAttribute("calYear", ym.getYear());
457 model.addAttribute("calMonth", ym.getMonthValue());
458 model.addAttribute("monthLabel", ym.getMonth().getDisplayName(TextStyle.FULL, Locale.ENGLISH) + " " + ym.getYear());
459 model.addAttribute("prevMonth", prev.toString());
460 model.addAttribute("nextMonth", next.toString());
461 model.addAttribute("shiftDates", shiftDatesList);
462 model.addAttribute("apptDates", apptDatesList);
463 model.addAttribute("surgeryDates", surgeryDatesList);
464 model.addAttribute("vacRanges", vacRanges);
465 model.addAttribute("pageTitle", "My Calendar");
466 model.addAttribute("currentPage", "calendar");
467 return "medical-specialist/calendar";
468 }

487
488 LocalDate sdTemp;
489 try {
490 sdTemp = (date != null && !date.isBlank()) ? LocalDate.parse(date) : LocalDate.now();
491 } catch (Exception e) {
492 sdTemp = LocalDate.now();
