13/**
14 * Thin wrapper around the Firebase Admin SDK and the Identity Toolkit REST API.
15 *
16 * Handles the three account-management operations the hospital system needs:
17 * creating staff accounts on hire, generating password-reset links, deleting
18 * accounts on termination, and verifying credentials on login.
19 *
20 * The Admin SDK is used for privileged server-side operations (create/delete/reset).
21 * Sign-in uses the public REST endpoint because the Admin SDK does not expose
22 * password verification — it is intended for server-side management, not auth.
23 *
24 * @see com.nvivx.vixhealthsystem.service.core.EmployeeService
25 * @see com.nvivx.vixhealthsystem.config.FirebaseConfig
26 */
27@Service
28public class FirebaseAuthService {

42 /**
43 * Creates a new Firebase Authentication account and returns its UID.
44 *
45 * If the email already exists (e.g., from a previous failed transaction) the
46 * existing account's UID is returned instead of throwing, so the caller can
47 * write the UID back to the database without restarting the whole flow.
48 *
49 * @param email Email address for the new account; must be unique.
50 * @param temporaryPassword Initial password that the employee should change on first login.
51 * @return Firebase UID of the created (or pre-existing) account.
52 * @throws Exception When Firebase is unreachable or returns an unexpected error.
53 */
54 public String createUser(String email, String temporaryPassword) throws Exception {
55 try {
56 UserRecord.CreateRequest request =
57 new UserRecord.CreateRequest()
58 .setEmail(email)
59 .setPassword(temporaryPassword)
60 .setEmailVerified(false)
61 .setDisabled(false);
62
63 UserRecord userRecord = FirebaseAuth.getInstance().createUser(request);
64 return userRecord.getUid();
65
66 } catch (com.google.firebase.auth.FirebaseAuthException e) {
67 if ("EMAIL_EXISTS".equals(e.getAuthErrorCode() != null
68 ? e.getAuthErrorCode().name() : "")) {
69 // Account already exists from a previous attempt — reuse the existing UID
70 return FirebaseAuth.getInstance().getUserByEmail(email).getUid();
71 }
72 // Check error message as fallback (some SDK versions use message instead of code)
73 if (e.getMessage() != null && e.getMessage().contains("EMAIL_EXISTS")) {
74 return FirebaseAuth.getInstance().getUserByEmail(email).getUid();
75 }
76 throw e;
77 }
78 }
79
80 /**
81 * Generates a Firebase password-reset link for the given email address.
82 *
83 * The link is valid for a fixed Firebase-determined window (typically 1 hour).
84 * In production this would be emailed to the user; in the demo it is logged to console.
85 *
86 * @param email The account's email address.
87 * @return One-time password-reset URL.
88 * @throws Exception When Firebase is unreachable or the email is not registered.
89 */
90 public String generatePasswordResetLink(String email) throws Exception {
91 return FirebaseAuth.getInstance().generatePasswordResetLink(email);
92 }
93
94 /**
95 * Permanently deletes a Firebase Authentication account.
96 *
97 * Called during employee termination ({@link com.nvivx.vixhealthsystem.service.core.EmployeeService#deleteEmployee}).
98 * If the UID is unknown or already deleted, Firebase throws and the caller handles it.
99 *
100 * @param firebaseUid The UID of the account to delete.
101 * @throws Exception When Firebase is unreachable or the UID does not exist.
102 */
103 public void deleteUser(String firebaseUid) throws Exception {
104 FirebaseAuth.getInstance().deleteUser(firebaseUid);
105 }

111 /**
112 * Verifies an email/password pair against Firebase and returns the user's Firebase UID.
113 *
114 * Uses the Identity Toolkit REST endpoint rather than the Admin SDK because the Admin
115 * SDK does not provide password verification — that is intentional by design.
116 * The returned UID is used to look up the matching {@link com.nvivx.vixhealthsystem.model.person.employee.Employee}
117 * row in the database.
118 *
119 * @param email The account's email address.
120 * @param password The password to verify.
121 * @return Firebase UID ({@code localId}) of the authenticated user.
122 * @throws Exception When credentials are wrong or Firebase is unreachable.
123 */
124 public String signInWithEmailAndPassword(String email, String password) throws Exception {
125 String url = "https://identitytoolkit.googleapis.com/v1/accounts:signInWithPassword?key="
126 + firebaseWebApiKey;
127
128 RestTemplate restTemplate = new RestTemplate();
129
130 Map<String, Object> request = new HashMap<>();
131 request.put("email", email);
132 request.put("password", password);
133 request.put("returnSecureToken", true);
134
135 ResponseEntity<Map> response = restTemplate.postForEntity(url, request, Map.class);
136
137 return (String) response.getBody().get("localId");
138 }
