15/**
16 * @brief Manages the lifecycle and status of medical equipment across all hospital facilities.
17 * Covers UC24 (machine malfunction reporting), FR5.3 (technician visibility), and
18 * FR5.4 (machine malfunction alerts).

20 * Annotated {@code @Transactional(readOnly=true)} at the class level; write methods override
21 * with {@code @Transactional}.

24 * ({@link #getFaultyMachines()}) and the domain-method path
25 * ({@link #getFaultyMachinesForTechnician(Technician)}), which applies the {@link Machinery}

133 /**
134 * Sets a machine's operational status and writes an audit entry (FR5.3).
135 *
136 * Also calls {@link Machinery#updateStatus()} in case the domain method carries
137 * additional side-effects (e.g., clearing maintenance timestamps).
138 * Adds an alert note in the audit message when the new status is {@link MachineStatus#FAULTY}.
139 *
140 * @param machineId ID of the machine to update.
141 * @param newStatus The target {@link MachineStatus}.
142 * @return The updated and re-persisted {@link Machinery}.
143 * @throws RuntimeException When no machine with {@code machineId} exists.
144 */
145 @Transactional
146 public Machinery updateMachineStatus(Long machineId, MachineStatus newStatus) {
147 Machinery machine = getMachineById(machineId);
148 MachineStatus oldStatus = machine.getStatus();
149
150 machine.setStatus(newStatus);
151
152 // Also call updateStatus() in case it has additional logic
153 machine.updateStatus();
154
155 Machinery saved = machineryRepository.save(machine);
156
157 // Log the status change
158 String alertMessage = "";
159 if (newStatus == MachineStatus.FAULTY) {
160 alertMessage = " ⚠️ ALERT: Machine malfunction detected!";
161 }
162
163 auditService.log("UPDATE_MACHINE_STATUS", "Machinery", String.valueOf(machineId),
164 "Status changed from " + oldStatus + " to " + newStatus + alertMessage);
165
166 return saved;
167 }

169 /**
170 * Marks a machine as {@link MachineStatus#FAULTY} and logs the issue (FR5.4).
171 *
172 * The fault alert becomes visible on the technician dashboard via {@link #getActiveAlerts()}.
173 *
174 * @param machineId ID of the machine that is malfunctioning.
175 * @param issueDescription Free-text description of the observed fault.
176 * @return The updated {@link Machinery} with FAULTY status.
177 * @throws RuntimeException When no machine with {@code machineId} exists.
178 */
179 @Transactional
180 public Machinery reportFaultyMachine(Long machineId, String issueDescription) {
181 Machinery machine = updateMachineStatus(machineId, MachineStatus.FAULTY);
182
183 auditService.log("REPORT_FAULTY_MACHINE", "Machinery", String.valueOf(machineId),
184 "Fault reported: " + issueDescription);
185
186 return machine;
187 }
188
189 /**
190 * Marks a machine as {@link MachineStatus#WORKING} after repair is complete.
191 *
192 * @param machineId ID of the repaired machine.
193 * @return The updated {@link Machinery} with WORKING status.
194 * @throws RuntimeException When no machine with {@code machineId} exists.
195 */
196 @Transactional
197 public Machinery repairMachine(Long machineId) {
198 Machinery machine = updateMachineStatus(machineId, MachineStatus.WORKING);
199
200 auditService.log("REPAIR_MACHINE", "Machinery", String.valueOf(machineId),
201 "Machine repaired and back to WORKING status");
202
203 return machine;
204 }

206 /**
207 * Transitions a machine to {@link MachineStatus#UNDER_MAINTENANCE} status.
208 *
209 * @param machineId ID of the machine to schedule for maintenance.
210 * @param reason Reason for the maintenance intervention.
211 * @return The updated {@link Machinery} with UNDER_MAINTENANCE status.
212 * @throws RuntimeException When no machine with {@code machineId} exists.
213 */
214 @Transactional
215 public Machinery scheduleMaintenance(Long machineId, String reason) {
216 Machinery machine = updateMachineStatus(machineId, MachineStatus.UNDER_MAINTENANCE);
217
218 auditService.log("SCHEDULE_MAINTENANCE", "Machinery", String.valueOf(machineId),
219 "Maintenance scheduled: " + reason);
220
221 return machine;
222 }

266 ))
267 .collect(Collectors.toList());
