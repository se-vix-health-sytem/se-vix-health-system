45 /**
46 * POST /contact/submit — log a contact-form submission and redirect with a confirmation.
47 *
48 * The message content is currently written to the application log only.
49 * A real deployment would forward the message to the hospital's support
50 * mailbox here instead. The redirect-after-POST pattern prevents duplicate
51 * submissions on browser reload.
52 *
53 * @param name Submitter's display name.
54 * @param email Submitter's email address for follow-up.
55 * @param message Free-text message body.
56 * @param redirectAttributes Flash attribute {@code successMessage} shown on the
57 * contact page after the redirect.
58 * @return Redirect to {@code /contact}.
59 */
60 @PostMapping("/contact/submit")
61 public String submitContact(@RequestParam String name,
62 @RequestParam String email,
63 @RequestParam String message,
64 RedirectAttributes redirectAttributes) {
65 log.info("Contact form submission from {} <{}>: {}", name, email, message);
66 redirectAttributes.addFlashAttribute("successMessage", "Thank you for your message! We'll get back to you soon.");
67 return "redirect:/contact";
68 }
