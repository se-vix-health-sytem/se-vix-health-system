15/**
16 * @brief Append-only JSON-file-backed repository for {@link AuditLog} entries.
17 *
18 * Persists audit logs to {@code src/main/resources/storage/audit-logs.json}.
19 * Intentionally exposes no {@code delete} or {@code update} methods to enforce
20 * NFR02 (Traceability) : once written, a log entry is immutable for normal users.
21 * The {@link #save(AuditLog)} method is {@code synchronized} to prevent ID
22 * collisions under concurrent request handling.
23 *
24 * @see com.nvivx.vixhealthsystem.model.AuditLog
25 */
26@Repository
27public class JsonAuditLogRepository {
28
29 // =========================================================
30 // STATE
31 // =========================================================
32
33 /** Jackson mapper configured to serialise dates as ISO-8601 strings. */
34 private final ObjectMapper mapper;
35
36 /** Relative path to the JSON backing file; resolved from the working directory. */
37 private final String path = "src/main/resources/storage/audit-logs.json";
38
39 // =========================================================
40 // CONSTRUCTOR
41 // =========================================================
42
43 /**
44 * Configures the mapper and ensures the backing file exists on disk.
45 *
46 * Creates an empty JSON array file if {@code audit-logs.json} is missing so
47 * that subsequent reads never fail with a file-not-found error.
48 */
49 public JsonAuditLogRepository() {
50 this.mapper = new ObjectMapper();
51 this.mapper.registerModule(new JavaTimeModule());
52 this.mapper.disable(com.fasterxml.jackson.databind.SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
53 initializeStorage();
54 }
55
56 // =========================================================
57 // INTERNAL HELPERS
58 // =========================================================
59
60 /**
61 * Creates the backing file with an empty array if it does not already exist.
62 *
63 * @throws RuntimeException if the file cannot be created
64 */
65 private void initializeStorage() {
66 File file = new File(path);
67 if (!file.exists()) {
68 try {
69 file.getParentFile().mkdirs();
70 mapper.writeValue(file, new ArrayList<AuditLog>());
71 } catch (IOException e) {
72 throw new RuntimeException("Could not create audit-logs.json", e);
73 }
74 }
75 }
76
77 // =========================================================
78 // READ OPERATIONS
79 // =========================================================
80
81 /**
82 * Returns all audit log entries in insertion order (NFR02 : read-only access).
83 *
84 * @return all log entries; never {@code null}, empty list if file is empty
85 * @throws RuntimeException if the file cannot be parsed
86 */
87 public List<AuditLog> findAll() {
88 try {
89 File file = new File(path);
90 if (!file.exists() || file.length() == 0) {
91 return new ArrayList<>();
92 }
93 return mapper.readValue(file, new TypeReference<List<AuditLog>>() {});
94 } catch (IOException e) {
95 throw new RuntimeException("Error reading audit-logs.json: " + e.getMessage(), e);
96 }
97 }
98
99 // =========================================================
100 // WRITE OPERATIONS (append-only)
101 // =========================================================
102
103 /**
104 * Appends a new log entry, assigning a sequential ID if absent (NFR02).
105 *
106 * Synchronised to prevent duplicate IDs under concurrent writes.
107 * Existing entries are never modified.
108 *
109 * @param log the entry to append; must not be {@code null}
110 * @return the same entry with its ID populated
111 * @throws RuntimeException if the file cannot be written
112 */
113 public synchronized AuditLog save(AuditLog log) {
114 List<AuditLog> allLogs = findAll();
115
116 // Generate ID if not present
117 if (log.getId() == null) {
118 long newId = allLogs.stream()
119 .mapToLong(AuditLog::getId)
120 .max()
121 .orElse(0L) + 1;
122 log.setId(newId);
123 }
124
125 allLogs.add(log);
126
127 try {
128 File file = new File(path);
129 mapper.writerWithDefaultPrettyPrinter().writeValue(file, allLogs);
130 } catch (IOException e) {
131 throw new RuntimeException("Error writing to audit-logs.json: " + e.getMessage(), e);
132 }
133
134 return log;
135 }
136
137 /**
138 * Finds all log entries for a given entity type.
139 *
140 * @param entityType case-sensitive entity name (e.g. {@code "Employee"}, {@code "Patient"})
141 * @return matching entries; empty list if none found
142 */
143 public List<AuditLog> findByEntityType(String entityType) {
144 return findAll().stream()
145 .filter(log -> entityType.equals(log.getEntityType()))
146 .collect(Collectors.toList());
147 }
148
149 /**
150 * Finds all log entries recorded under a specific username.
151 *
152 * @param username the actor's username as stored in the log entries
153 * @return matching entries; empty list if none found
154 */
155 public List<AuditLog> findByUsername(String username) {
156 return findAll().stream()
157 .filter(log -> username.equals(log.getUsername()))
158 .collect(Collectors.toList());
159 }
160
161 /**
162 * Finds all log entries whose action field contains the given keyword.
163 *
164 * @param action action keyword (e.g. {@code "CREATE"}, {@code "UPDATE"}, {@code "DELETE"})
165 * @return matching entries; empty list if none found
166 */
167 public List<AuditLog> findByAction(String action) {
168 return findAll().stream()
169 .filter(log -> log.getAction() != null && log.getAction().contains(action))
170 .collect(Collectors.toList());
171 }
172
173 /**
174 * Returns the most recent log entries sorted by timestamp descending.
175 *
176 * @param limit maximum number of entries to return; must be positive
177 * @return up to {@code limit} most recent entries
178 */
179 public List<AuditLog> findRecent(int limit) {
180 return findAll().stream()
181 .sorted((a, b) -> b.getTimestamp().compareTo(a.getTimestamp()))
182 .limit(limit)
183 .collect(Collectors.toList());
184 }

188}
