35 /**
36 * GET /about — render the public About page.
37 *
38 * Department count is computed by de-duplicating on lowercase name so that
39 * departments sharing a name across different facilities are not double-counted
40 * in the displayed statistic.
41 *
42 * @param model Receives {@code totalSpecialists}, {@code totalDepartments},
43 * {@code totalEmployees}, and {@code pageTitle} attributes.
44 * @return Thymeleaf template {@code site/about}.
45 */
46 @GetMapping("/about")
47 public String about(Model model) {
48 model.addAttribute("totalSpecialists", employeeService.findAllMedicalSpecialists().size());
49 long distinctDepts = departmentService.getAllDepartments().stream()
50 .map(d -> d.getName().toLowerCase())
51 .distinct()
52 .count();
53 model.addAttribute("totalDepartments", distinctDepts);
54 model.addAttribute("totalEmployees", employeeService.getTotalEmployeeCount());
55 model.addAttribute("pageTitle", "About VIX Health System");
56 return "site/about";
57 }
