15/**
16 * @brief Manages appointment booking, rescheduling, and cancellation for the VIX Health System.
17 *
18 * Appointments are stored in a JSON file via {@link JsonAppointmentRepository} rather than
19 * a relational table, so there is no {@code @Transactional} support; all reads and writes
20 * are synchronized by the repository implementation.
21 *
22 * Slot-conflict detection is enforced before every new booking to prevent double-booking
23 * across the shared appointment list.
24 *
25 * @see com.nvivx.vixhealthsystem.model.medical.Appointment

55 /**
56 * Books a new appointment from a structured request DTO, checking for slot conflicts.
57 *
58 * @param req Appointment details including date-time, duration, and notes.
59 * @return The persisted {@link Appointment} with a generated ID.

69 req.getDateTime(),
70 req.getDuration(),
71 req.getNotes()
72 );
73
74 all.add(appt);
75 repo.saveAll(all);

80 /**
81 * Books an appointment on behalf of a patient through a secretary (UC-Secretary flow).
82 *
83 * Delegates to {@link Secretary#makeAppointmentForPatient} so that domain invariants
84 * (e.g., patient eligibility) are enforced by the model. The slot-conflict guard
85 * runs before the domain call to fail fast.
86 *
87 * @param secretary The acting secretary; must not be {@code null}.
88 * @param patient The patient for whom the appointment is being booked.
89 * @param specialist The medical specialist who will conduct the appointment.
90 * @param dt Requested date and time for the appointment.
91 * @param duration Duration in minutes.
92 * @param notes Optional clinical notes; may be {@code null}.
93 * @return The persisted {@link Appointment}.
94 * @throws SlotNotAvailableException When the slot overlaps an existing booking.
95 */
96 public Appointment bookForSecretary(Secretary secretary,
97 Patient patient,
98 MedicalSpecialist specialist,
99 LocalDateTime dt,
100 int duration,
101 String notes) {
102 List<Appointment> all = repo.findAll();
103 checkConflict(all, dt, duration);
104
105 // Domain: secretary creates appointment for patient via model methods
106 Appointment appt = secretary.makeAppointmentForPatient(patient, specialist, dt);
107 appt.setId((int) System.currentTimeMillis());
108 appt.setDuration(duration);
109 appt.setNotes(notes);
110 appt.setStatus("PENDING");
111 appt.setPaymentStatus(false);
112
113 all.add(appt);
114 repo.saveAll(all);
115
116 return appt;
117 }

119 /**
120 * Cancels an appointment, delegating cancellability checks to the domain model.
121 *
122 * @param appointmentId ID of the appointment to cancel.
123 * @return The updated {@link Appointment} with canceled status.
124 * @throws RuntimeException When no appointment with {@code appointmentId} exists.
125 */
126 public Appointment cancelAppointment(int appointmentId) {
127 List<Appointment> all = repo.findAll();
128 Appointment appt = all.stream()
129 .filter(a -> a.getId() == appointmentId)
130 .findFirst()
131 .orElseThrow(() -> new RuntimeException("Appointment not found: " + appointmentId));
132
133 appt.cancel(); // domain method — enforces isCancellable()
134 repo.saveAll(all);
135 return appt;
136 }

138 /**
139 * Reschedules an active appointment, delegating validity checks to the domain model.
140 *
141 * @param appointmentId ID of the appointment to reschedule.
142 * @param newDateTime The new date and time; must be in the future.
143 * @return The updated {@link Appointment} with the new date-time.
144 * @throws RuntimeException When no appointment with {@code appointmentId} exists.
145 */
146 public Appointment rescheduleAppointment(int appointmentId, LocalDateTime newDateTime) {
147 List<Appointment> all = repo.findAll();
148 Appointment appt = all.stream()
149 .filter(a -> a.getId() == appointmentId)
150 .findFirst()
151 .orElseThrow(() -> new RuntimeException("Appointment not found: " + appointmentId));
152
153 appt.reschedule(newDateTime); // domain method — enforces isActive()
154 repo.saveAll(all);
155 return appt;
156 }

171 /**
172 * Throws {@link SlotNotAvailableException} if {@code newStart + duration} overlaps
173 * any existing appointment.
174 *
175 * Two intervals overlap when {@code newStart < existingEnd && newEnd > existingStart}.
176 *
177 * @param existing Current list of all appointments.
178 * @param newStart Requested start date-time for the new booking.
179 * @param duration Duration of the new booking in minutes.
180 * @throws SlotNotAvailableException When at least one overlap is detected.
181 */
182 private void checkConflict(List<Appointment> existing, LocalDateTime newStart, int duration) {
183 LocalDateTime newEnd = newStart.plusMinutes(duration);
184 boolean conflict = existing.stream().anyMatch(a -> {
185 LocalDateTime existingStart = a.getDateTime();
186 LocalDateTime existingEnd = existingStart.plusMinutes(a.getDuration());
187 return newStart.isBefore(existingEnd) && newEnd.isAfter(existingStart);
188 });
189 if (conflict) {
190 throw new SlotNotAvailableException("This slot is already booked");
191 }
192 }
