87 /**
88 * Searches patients by name, surname, or fiscal code using case-insensitive
89 * substring matching.
90 *
91 * Filtering is performed in-memory because the repository does not yet expose a
92 * compound search query. A blank query returns all patients.
93 *
94 * @param query Search term; {@code null} or blank returns all patients.
95 * @return Non-null list of matching patients.
96 */
97 public List<Patient> searchPatients(String query) {
98 if (query == null || query.trim().isEmpty()) {
99 return findAllPatients();
100 }
101
102 String searchLower = query.toLowerCase().trim();
103
104 return patientRepository.findAll().stream()
105 .filter(patient ->
106 (patient.getName() != null && patient.getName().toLowerCase().contains(searchLower)) ||
107 (patient.getSurname() != null && patient.getSurname().toLowerCase().contains(searchLower)) ||
108 (patient.getFiscalCode() != null && patient.getFiscalCode().toLowerCase().contains(searchLower))
109 )
110 .collect(Collectors.toList());
111 }

156 /**
157 * Anonymises a patient account while preserving all clinical records (GDPR soft-delete).
158 *
159 * Personal identifiers (name, surname, email, phone) are overwritten or nulled.
160 * The fiscal code is replaced with a timestamped tombstone token so the row remains
161 * uniquely identifiable internally. Medical conditions, prescriptions, and surgeries
162 * are intentionally kept to satisfy the 5-year clinical retention requirement (NFR09).
163 *
164 * Prefer this method over {@link #permanentDeletePatient(long)} for all normal deletion
165 * requests.
166 *
167 * Deletes (anonymizes) a patient account while preserving medical records.
168 *
169 * GDPR Compliance (NFR04):
170 * - Personal data (name, surname, email, phone) is removed
171 * - Fiscal code is replaced with a unique DELETED_ timestamp
172 * - Medical records (conditions, prescriptions, surgeries) are preserved
173 * for clinical and legal requirements (NFR09 - 5-year retention)
174 */
175 @Transactional
176 public void deletePatient(long id) {
177 Patient patient = findById(id);
178
179 // Store original name for audit log
180 String originalName = patient.getName() + " " + patient.getSurname();
181
182 // Domain: patient clears their own transient state (appointments list, medical record ref)
183 patient.deleteAccount();
184
185 // Anonymize persistent personal data (GDPR compliance)
186 patient.setName("ANONYMIZED");
187 patient.setSurname("ANONYMIZED");
188 patient.setEmail(null);
189 patient.setPhoneNumber(null);
190 patient.setFiscalCode("DELETED_" + System.currentTimeMillis());
191
192 // Note: birthDate, birthPlace, gender are kept as they are
193 // not considered direct identifiers when name is removed
194
195 // Save the anonymized patient record
196 Patient saved = patientRepository.save(patient);
197
198 // Log the deletion (anonymization)
199 auditService.log("DELETE_PATIENT", "Patient", String.valueOf(id),
200 "Patient account anonymized. Original: " + originalName);
201
202 // Note: Medical records remain in the database with patient_id = id
203 // but the patient is no longer identifiable. This satisfies:
204 // - NFR04 (GDPR) - personal data is removed
205 // - NFR09 (Medical Data Norms) - clinical records retained for 5 years
206 }
207
208 /**
209 * Permanently deletes a patient and ALL associated medical records from the database.
210 *
211 * This is a destructive, non-recoverable operation intended only for testing environments
212 * or court-ordered erasure when anonymisation is legally insufficient. In all other
213 * cases, use {@link #deletePatient(long)} instead.
214 *
215 * @param id ID of the patient to permanently erase.
216 * @throws RuntimeException When no patient with the given ID exists.
217 */
218 @Transactional
219 public void permanentDeletePatient(long id) {
220 Patient patient = findById(id);
221 String patientInfo = patient.getName() + " " + patient.getSurname();
222
223 patientRepository.delete(patient);
224
225 auditService.log("PERMANENT_DELETE_PATIENT", "Patient", String.valueOf(id),
226 "Permanently deleted patient: " + patientInfo + " and ALL medical records");
227 }
