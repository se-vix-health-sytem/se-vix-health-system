20/**
21 * @brief Manages patient medical records — creation and appending of diagnoses,
22 * prescriptions, and exam results.
23 *
24 * Annotated {@code @Transactional(readOnly=true)} at the class level; write methods
25 * override with {@code @Transactional}.
26 *
27 * All mutations are recorded via {@link AuditService} to satisfy NFR02 (traceability).
28 * When a patient has no record yet, write operations auto-create one rather than failing.

30 * Prescription issuance delegates to {@link com.nvivx.vixhealthsystem.model.person.employee.MedicalSpecialist#appPrescriptionForPatient}

33 * @see PatientService

77 /**
78 * Returns the medical record for the given patient, throwing when absent.
79 *
80 * @param patientId ID of the patient.
81 * @return The patient's {@link MedicalRecord}; never {@code null}.
82 * @throws RuntimeException When no medical record exists for the patient.

84 public MedicalRecord getMedicalRecordByPatientId(Long patientId) {

89 /**
90 * Loads a patient and touches their medical record proxy to force initialisation.
91 *
92 * JPA lazy-loading means the record may still be a proxy after {@code findById}; this
93 * method ensures the proxy is initialised before the session closes.
94 *
95 * @param patientId ID of the patient to load.
96 * @return The {@link Patient} with the {@code medicalRecord} association initialised.
97 * @throws RuntimeException When no patient with {@code patientId} exists.
98 */
99 public Patient getPatientWithMedicalRecord(Long patientId) {
100 Patient patient = patientRepository.findById(patientId)
101 .orElseThrow(() -> new RuntimeException("Patient not found: " + patientId));
102
103 // Force load medical record if needed (JPA will handle lazy loading)
104 if (patient.getMedicalRecord() != null) {
105 patient.getMedicalRecord().getId(); // Touch to initialize if proxy
106 }
107
108 return patient;
109 }

115 /**
116 * Creates and links a new {@link MedicalRecord} for a patient.
117 *
118 * All three metric parameters are optional; pass {@code null} when not yet known.
119 * Allergies and vaccines are initialised to empty strings to avoid null-checks downstream.
120 *
121 * @param patientId ID of the patient who owns the record.
122 * @param height Patient height in centimetres; may be {@code null}.
123 * @param weight Patient weight in kilograms; may be {@code null}.
124 * @param bloodType ABO+Rh blood group string (e.g., {@code "A+"}); may be {@code null}.
125 * @return The persisted {@link MedicalRecord} with a generated ID.
126 * @throws RuntimeException When no patient with {@code patientId} exists.
127 */
128 @Transactional
129 public MedicalRecord createMedicalRecord(Long patientId, Float height, Float weight, String bloodType) {
130 Patient patient = patientRepository.findById(patientId)
131 .orElseThrow(() -> new RuntimeException("Patient not found: " + patientId));
132
133 MedicalRecord record = new MedicalRecord(height, weight, bloodType);
134 record.setPatient(patient);
135 record.setAllergies("");
136 record.setVaccines("");
137
138 MedicalRecord saved = medicalRecordRepository.save(record);
139 patient.setMedicalRecord(saved);
140 patientRepository.save(patient);
141
142 auditService.log("CREATE_MEDICAL_RECORD", "MedicalRecord", String.valueOf(saved.getId()),
143 "Created medical record for patient: " + patientId);
144
145 return saved;
146 }

148 /**
149 * Appends a diagnosis to the patient's medical record, creating the record first
150 * if it does not exist.
151 *
152 * @param patientId ID of the patient to diagnose.
153 * @param diagnosisName Short clinical label for the condition.
154 * @param description Detailed clinical notes.
155 * @param severity Severity tier (e.g., {@code "MILD"}, {@code "SEVERE"}).
156 * @throws RuntimeException When no patient with {@code patientId} exists.
157 */
158 @Transactional
159 public void addDiagnosis(Long patientId, String diagnosisName, String description, String severity) {
160 Patient patient = getPatientWithMedicalRecord(patientId);
161 MedicalRecord record = patient.getMedicalRecord();
162
163 if (record == null) {
164 record = createMedicalRecord(patientId, null, null, null);
165 }
166
167 MedicalCondition condition = new MedicalCondition();
168 condition.setName(diagnosisName);
169 condition.setDescription(description);
170 condition.setDateOfDiagnosis(LocalDate.now());
171 condition.setType(severity);
172
173 record.addCondition(condition);
174 medicalRecordRepository.save(record);
175
176 auditService.log("ADD_DIAGNOSIS", "MedicalRecord", String.valueOf(record.getId()),
177 "Added diagnosis: " + diagnosisName + " (severity: " + severity + ") for patient: " + patientId);
178 }

180 /**
181 * Issues a prescription for the patient via the prescribing specialist's domain method.
182 *
183 * Delegates to {@link com.nvivx.vixhealthsystem.model.person.employee.MedicalSpecialist#appPrescriptionForPatient}
184 * so that the specialist back-reference on the {@link Prescription} is set correctly.
185 * Auto-creates a medical record when the patient has none.
186 *
187 * @param patientId ID of the patient receiving the prescription.
188 * @param medicalSpecialistId ID of the issuing {@link MedicalSpecialist}.
189 * @param medication Name or description of the prescribed medication.
190 * @throws RuntimeException When the patient or specialist cannot be found, or the
191 * employee ID does not resolve to a {@link MedicalSpecialist}.
192 */
193 @Transactional
194 public void addPrescription(Long patientId, Long medicalSpecialistId, String medication) {
195 Patient patient = getPatientWithMedicalRecord(patientId);
196 MedicalSpecialist specialist = (MedicalSpecialist) employeeRepository.findById(medicalSpecialistId)
197 .orElseThrow(() -> new RuntimeException("Medical specialist not found: " + medicalSpecialistId));
198
199 MedicalRecord record = patient.getMedicalRecord();
200
201 if (record == null) {
202 record = createMedicalRecord(patientId, null, null, null);
203 // Reload patient so the new record is attached to the same in-memory object
204 patient = patientRepository.findById(patientId)
205 .orElseThrow(() -> new RuntimeException("Patient not found: " + patientId));
206 }
207
208 Prescription prescription = new Prescription();
209 prescription.setMedication(medication);
210 prescription.setDateTime(LocalDateTime.now());
211
212 // Domain: specialist issues prescription for patient via model method,
213 // which sets the specialist back-reference and adds to the medical record
214 specialist.appPrescriptionForPatient(patient, prescription);
215 medicalRecordRepository.save(patient.getMedicalRecord());
216
217 auditService.log("ADD_PRESCRIPTION", "MedicalRecord", String.valueOf(record.getId()),
218 "Added prescription: " + medication + " for patient: " + patientId + " by specialist: " + medicalSpecialistId);
219 }

221 /**
222 * Records an exam result in the patient's medical record.
223 *
224 * Exam results are modelled as {@link MedicalCondition} entries with
225 * {@code type = "EXAM_RESULT"} because no separate exam-result entity exists yet.
226 * Auto-creates a medical record when the patient has none.
227 *
228 * @param patientId ID of the patient.
229 * @param examType Type of examination (e.g., {@code "Blood Test"}, {@code "MRI"}).
230 * @param result The outcome of the exam (e.g., {@code "Normal"}, numeric values).
231 * @param notes Optional additional clinical notes; may be {@code null} or blank.
232 * @throws RuntimeException When no patient with {@code patientId} exists.
233 */
234 @Transactional
235 public void addExamResult(Long patientId, String examType, String result, String notes) {
236 Patient patient = getPatientWithMedicalRecord(patientId);
237 MedicalRecord record = patient.getMedicalRecord();
238
239 if (record == null) {
240 record = createMedicalRecord(patientId, null, null, null);
241 }
242
243 String description = "Result: " + result;
244 if (notes != null && !notes.isEmpty()) {
245 description += "\nNotes: " + notes;
246 }
247
248 MedicalCondition examResult = new MedicalCondition();
249 examResult.setName(examType);
250 examResult.setDescription(description);
251 examResult.setDateOfDiagnosis(LocalDate.now());
252 examResult.setType("EXAM_RESULT");
253
254 record.addCondition(examResult);
255 medicalRecordRepository.save(record);
256
257 auditService.log("ADD_EXAM_RESULT", "MedicalRecord", String.valueOf(record.getId()),
258 "Added exam result: " + examType + " for patient: " + patientId);
259
260 }
