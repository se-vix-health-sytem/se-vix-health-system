13/**
14 * @brief Controller for the public hospital-location map pages — base URL {@code /map}.
15 *
16 * Provides the two location-related views: an interactive map that pins all
17 * VIX Health System facilities, and a directions page that helps visitors plan
18 * their journey. Facility coordinates are hard-coded to match the seeded
19 * {@code MedicalFacilities} database rows; the data is serialised into plain
20 * {@code Map} objects so Thymeleaf's inline JavaScript engine produces valid
21 * JS literals without any extra serialisation step.
22 */

31 /**
32 * GET /map/hospitals — render the interactive hospital-location map.
33 *
34 * Each facility is projected into a plain {@link Map} ({@code name}, {@code lat},
35 * {@code lng}, {@code address}, {@code phone}, {@code type}) so that Thymeleaf's
36 * {@code th:inline="javascript"} block can serialise the list directly as a JS
37 * array without a custom converter.
38 *
39 * @param model Receives {@code locations} (list of plain maps) and {@code pageTitle}.
40 * @return Thymeleaf template {@code site/map/hospitals}.
41 */
42 @GetMapping("/hospitals")
43 public String showHospitalMap(Model model) {
44 // Convert to List<Map> so Thymeleaf's JS inline serializer produces valid JS objects
45 List<Map<String, Object>> locations = getHospitalLocations().stream()
46 .map(loc -> {
47 Map<String, Object> m = new HashMap<>();
48 m.put("name", loc.getName());
49 m.put("lat", loc.getLat());
50 m.put("lng", loc.getLng());
51 m.put("address", loc.getAddress());
52 m.put("phone", loc.getPhone());
53 m.put("type", loc.getType());
54 return m;
55 })
56 .collect(java.util.stream.Collectors.toList());
57 model.addAttribute("locations", locations);
58 model.addAttribute("pageTitle", "Find Our Locations");
59 return "site/map/hospitals";
60 }
61
