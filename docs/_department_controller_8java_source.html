11/**
12 * @brief Controller for the public department directory — base URL {@code /departments}.
13 *
14 * Exposes a browseable listing of all clinical departments and a detail page
15 * for each one, showing the assigned doctors, the services offered, and a
16 * profile image map. The Administration department is hidden from the public
17 * list because it has no patient-facing services.
18 *
19 * @see DepartmentService
20 */
21@Controller
22@RequestMapping("/departments")
23public class DepartmentController {
24
25 private final DepartmentService departmentService;
26
27 public DepartmentController(DepartmentService departmentService) {
28 this.departmentService = departmentService;
29 }

35 /**
36 * GET /departments — list all patient-facing clinical departments.
37 *
38 * The Administration department is excluded because it is internal-only
39 * and should not appear in the patient-facing section of the site.
40 *
41 * @param model Receives {@code departments} (filtered list) and {@code pageTitle}.
42 * @return Thymeleaf template {@code departments/list}.
43 */
44 @GetMapping
45 public String listDepartments(Model model) {
46 var departments = departmentService.getAllDepartments().stream()
47 .filter(d -> !d.getName().equalsIgnoreCase("Administration"))
48 .toList();
49 model.addAttribute("departments", departments);
50 model.addAttribute("pageTitle", "All Departments");
51 return "departments/list";
52 }
53
54 /**
55 * GET /departments/{id} — display the detail page for a single department.
56 *
57 * Loads the department by its string identifier, then separately queries
58 * the doctors, services, and doctor image map that belong to it. Using
59 * the department's numeric {@code Long} ID for the service calls avoids
60 * exposing the surrogate key in the URL path.
61 *
62 * @param id String identifier of the department (as used in URLs).
63 * @param model Receives {@code department}, {@code doctors}, {@code services},
64 * {@code doctorImages}, and {@code pageTitle} attributes.
65 * @return Thymeleaf template {@code departments/detail}, or
66 * {@code redirect:/departments} when the identifier is not found.
67 */
68 @GetMapping("/{id}")
69 public String departmentDetail(@PathVariable String id, Model model) {
70 Department department = departmentService.getDepartmentById(id);
71 if (department == null) {
72 return "redirect:/departments";
73 }
74 Long deptId = department.getId();
75 model.addAttribute("department", department);
76 model.addAttribute("doctors", departmentService.getDoctorsByDepartment(deptId));
77 model.addAttribute("services", departmentService.getServicesByDepartment(deptId));
78 model.addAttribute("doctorImages", departmentService.getDoctorImageMap(deptId));
79 model.addAttribute("pageTitle", department.getName());
80 return "departments/detail";
81 }
82}
